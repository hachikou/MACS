253  public DBTable AddColumns(params string[] cols) {
254  if(cols == null)
255  return this;
256  string[] newcolumns;
257  string[] newsqlcolumns;
258  int idx;
259  if(columns == null) {
260  newcolumns = new string[cols.Length];
261  newsqlcolumns = new string[cols.Length];
262  idx = 0;
263  } else {
264  newcolumns = new string[columns.Length+cols.Length];
265  newsqlcolumns = new string[columns.Length+cols.Length];
266  for(int i = 0; i < columns.Length; i++){
267  newcolumns[i] = columns[i];
268  newsqlcolumns[i] = (sqlcolumns==null)?columns[i]:sqlcolumns[i];
269  }
270  idx = columns.Length;
271  }
272  for(int i = 0; i < cols.Length; i++) {
273  string c = cols[i];
274  int pos = c.IndexOf(" as ", StringComparison.OrdinalIgnoreCase);
275  if(pos < 0) {
276  newcolumns[idx+i] = c;
277  newsqlcolumns[idx+i] = c;
278  } else {
279  newcolumns[idx+i] = c.Substring(pos+4);
280  newsqlcolumns[idx+i] = c;
281  }
282  }
283  columns = newcolumns;
284  sqlcolumns = newsqlcolumns;
285  return this;
286  }

355  /// </para>
356  /// </remarks>
357  public DBTable AddAllJoinColumnsUniqName(string joinname) {
358  if(joinList == null)
359  throw new ArgumentException("This DBTable does not have joined tables");
360  foreach(JoinTable jt in joinList) {
361  if(jt.name == joinname) {
362  string[] jcolumns;
363  if(jt.def != null)
364  jcolumns = jt.def.ColumnNames;
365  else if(jt.tbl != null)
366  jcolumns = jt.tbl.Columns;
367  else
368  throw new ArgumentException("This joined table does not have valid column definition");
369  string[] newcolumns = new string[jcolumns.Length];
370  for(int i = 0; i < jcolumns.Length; i++) {
371  bool duplicated = false;
372  string[] currentColumns = Columns;
373  for(int j = 0; j < currentColumns.Length; j++) {
374  if(currentColumns[j] == jcolumns[i]) {
375  duplicated = true;
376  break;
377  }
378  }
379  if(duplicated)
380  newcolumns[i] = joinname+"."+jcolumns[i];
381  else
382  newcolumns[i] = jcolumns[i];
383  }
384  return AddColumns(newcolumns);
385  }
386  }
387  throw new ArgumentException(String.Format("Table {0} is not joined to this table.", joinname));

425  /// <summary>
426  /// 指定した名前のカラムの定義を獲得する
427  /// </summary>
428  /// <remarks>
429  /// <para>
430  /// "テーブル名.カラム名"の形式の時には、Joinしているテーブルも探して
431  /// カラム定義を獲得する。
432  /// </para>
433  /// </remarks>
434  public DBColumnDef GetColumnDef(string colname) {
435  int idx = colname.IndexOf(" as ", StringComparison.OrdinalIgnoreCase);
436  if(idx >= 0) {
437  colname = colname.Substring(0, idx);
438  }
439  idx = colname.IndexOf('.');
440  if(idx >= 0) {
441  string tblname = colname.Substring(0,idx);
442  colname = colname.Substring(idx+1);
443  if(tblname == name) {
444  if(def != null)
445  return def.GetColumnDef(colname);
446  else if(tbl != null)
447  return tbl.GetColumnDef(colname);
448  }
449  if(joinList == null)
450  return null;
451  foreach(JoinTable jt in joinList) {
452  if(tblname == jt.name) {
453  if(jt.def != null)
454  return jt.def.GetColumnDef(colname);
455  else if(jt.tbl != null)
456  return jt.tbl.GetColumnDef(colname);
457  }
458  }
459  return null;
460  }
461  if(joinList != null) {
462  if((colname == "rowid") || ((def != null) && def.HasColumn(colname)) || ((tbl != null) && tbl.ContainsColumn(colname))) {
463  if(def != null)
464  return def.GetColumnDef(colname);
465  else if(tbl != null)
466  return tbl.GetColumnDef(colname);
467  }
468  foreach(JoinTable jt in joinList) {
469  if((jt.def != null) && jt.def.HasColumn(colname))
470  return jt.def.GetColumnDef(colname);
471  else if((jt.tbl != null) && jt.tbl.ContainsColumn(colname))
472  return jt.tbl.GetColumnDef(colname);
473  }
474  }
475  if(def != null)
476  return def.GetColumnDef(colname);
477  else if(tbl != null)
478  return tbl.GetColumnDef(colname);
479  else
480  throw new ArgumentException("DBTable or DBTableDef must not be null.");
481  }
