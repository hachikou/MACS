177  private void _open(Type dbtype_, string server, string dbname_, string user, string passwd) {
178  dbtype = dbtype_;
179  dbname = dbname_;
180  if(String.IsNullOrEmpty(dbname))
181  throw new ArgumentException("No Database name");
182  if(dbname == DEFAULT_DATABASE) {
183  switch(dbtype) {
184 #if USE_MYSQL
185  case Type.MySQL:
186  dbname = "mysql";
187  break;
188 #endif
189 #if USE_POSTGRESQL
190  case Type.PostgreSQL:
191  dbname = "postgres";
192  break;
193 #endif
194  default:
195  dbname = "";
196  break;
197  }
198  }
199  StringBuilder constr = new StringBuilder();
200  switch(dbtype) {
201 #if USE_MYSQL
202  case Type.MySQL:
203  constr.Append("Server=");
204  constr.Append(server);
205  constr.Append(";Database=");
206  constr.Append(dbname);
207  constr.Append(";Uid=");
208  constr.Append(user);
209  constr.Append(";Pwd=");
210  constr.Append(passwd);
211  constr.Append(";CharSet=utf8");
212  break;
213 #endif
214 #if USE_ORACLE
215  case Type.Oracle:
216  constr.Append("User ID=");
217  constr.Append(user);
218  constr.Append(";Password=");
219  constr.Append(passwd);
220  constr.Append(";Data Source=(DESCRIPTION=(ADDRESS=(PROTOCOL=TCP)(HOST=");
221  constr.Append(server);
222  constr.Append(")(PORT=1521))(CONNECT_DATA=(SERVER=DEDICATED)(SERVICE_NAME=");
223  constr.Append(dbname);
224  constr.Append(")))");
225  break;
226 #endif
227 #if USE_POSTGRESQL
228  case Type.PostgreSQL:
229  constr.Append("Server=");
230  constr.Append(server);
231  constr.Append(";Port=5432;User Id=");
232  constr.Append(user);
233  constr.Append(";Password=");
234  constr.Append(passwd);
235  constr.Append(";Pooling=False");
236  constr.Append(";Encoding=UNICODE");
237  constr.Append(";Database=");
238  constr.Append(dbname);
239  break;
240 #endif
241  default:
242  // go through.
243  break;
244  }
245 
246  //log("Open "+constr.ToString());
247  switch(dbtype) {
248 #if USE_MYSQL
249  case Type.MySQL:
250  con = new MySqlConnection(constr.ToString());
251  break;
252 #endif
253 #if USE_ORACLE
254  case Type.Oracle:
255  con = new OracleConnection(constr.ToString());
256  break;
257 #endif
258 #if USE_POSTGRESQL
259  case Type.PostgreSQL:
260  con = new NpgsqlConnection(constr.ToString());
261  break;
262 #endif
263  default:
264  throw new ArgumentException("Unsupported DB type");
265  }
266  con.Open();
267  fetchConnectionId();
268  }

428  /// <summary>
429  /// Query発行
430  /// </summary>
431  public DBReader Query(string sql) {
432  if(con == null)
433  return new DBReader(); // 何を読み取ってもNULLになるダミーリーダ
434  lock(con) {
435  log(sql);
436  if(reader != null) {
437  reader.Close();
438  reader = null;
439  }
440  using(DbCommand cmd = con.CreateCommand()) {
441  if(commandtimeout > 0)
442  cmd.CommandTimeout = commandtimeout;
443  cmd.CommandText = sql;
444  try {
445  currentSQL = sql;
446  timer.Restart();
447  reader = new DBReader(cmd.ExecuteReader());
448  timer.Stop();
449  if(timer.ElapsedMilliseconds >= WarnThresholdTime)
450  logWarning(String.Format("Heavy SQL: {0}msec: SQL={1}", timer.ElapsedMilliseconds, sql));
451  return reader;
452  } catch(DbException e) {
453  logErr(String.Format("SQL Error: {0}: SQL={1}", e.Message, sql));
454  if(ignoreexception)
455  return new DBReader(); // 握りつぶすときはダミーリーダー
456  else
457  throw;
458  } catch(Exception) {
459  abort();
460  if(ignoreexception)
461  return new DBReader(); // 握りつぶすときはダミーリーダー
462  else
463  throw;
464  } finally {
465  currentSQL = null;
466  }
467  }
468  }
469  }

637  private int _execute(string sql) {
638  log(sql);
639  using(DbCommand cmd = con.CreateCommand()) {
640  if(commandtimeout > 0)
641  cmd.CommandTimeout = commandtimeout;
642  cmd.CommandText = sql;
643  try {
644  currentSQL = sql;
645  timer.Restart();
646  int ret = cmd.ExecuteNonQuery();
647  timer.Stop();
648  if(timer.ElapsedMilliseconds >= WarnThresholdTime)
649  logWarning(String.Format("Heavy SQL: {0}msec: SQL={1}", timer.ElapsedMilliseconds, sql));
650  return ret;
651  } catch(DbException e) {
652  logErr(String.Format("SQL Error: {0}: SQL={1}", e.Message, sql));
653  if(ignoreexception)
654  return -1; // 握りつぶすときは -1
655  else
656  throw;
657  } catch(Exception) {
658  abort();
659  if(ignoreexception)
660  return -1; // 握りつぶすときは -1
661  else
662  throw;
663  } finally {
664  currentSQL = null;
665  }

790  /// <summary>
791  /// SQLを実行して実行結果をテキストストリームにHumanReadableな形で書き出す
792  /// </summary>
793  public void Execute(string sql, TextWriter w) {
794  try {
795  if(sql.StartsWith("select", StringComparison.InvariantCultureIgnoreCase)) {
796  List<string[]> values = new List<string[]>();
797  string[] columns;
798  int[] fieldLen;
799  using(DBReader reader = Query(sql)) {
800  columns = reader.GetColumns();
801  fieldLen = new int[columns.Length];
802  for(int i = 0; i < fieldLen.Length; i++){

823  w.Write(' ');

825  }
826  w.WriteLine('|');
827  for(int i = 0; i < fieldLen.Length; i++) {
828  w.Write('|');
829  for(int j = 0; j < fieldLen[i]+2; j++)
830  w.Write('-');
831  }
832  w.WriteLine('|');

834  for(int i = 0; i < fieldLen.Length; i++) {
835  w.Write('|');

837  w.Write(' ');

840  w.Write(' ');
