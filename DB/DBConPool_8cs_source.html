216  internal DbConnection GetDbConnection() {
217  lock(mutex) {
218  // まず有効期限切れDBConをDisposeする
219  purge(false);
220  // 空いているDBConを探す
221  foreach(Pool p in pool) {
222  if(p.dbcon == null) {
223  p.dbcon = new DBCon(DBType, Server, DBName, User, Passwd);
224  p.busy = true;
225  p.dbcon.CommandTimeout = CommandTimeout;
226  return p.dbcon.DBConnection;
227  } else if(!p.busy) {
228  p.busy = true;
229  return p.dbcon.DBConnection;
230  }
231  }
232  LOG_ERR("All DBCon pool ({0}) has been exhausted.", MaxPoolSize);
233  dumpStatus(this.Logger, null);
234  throw new InvalidOperationException("No more DBCon pool.");
235  }
236  }

238  internal void Free(DbConnection con) {
239  lock(mutex) {
240  for(int i = 0; i < pool.Length; i++) {
241  Pool p = pool[i];
242  if((p.dbcon != null) && (p.dbcon.DBConnection == con)) {
243  NThread thread = NThread.CurrentThread;
244  if((thread != null) && thread.IsAborting) {
245  p.dbcon.Dispose();
246  p.dbcon = null;
247  }
248  p.busy = false;
249  p.timer.Restart();
250  break;
251  }
252  }
253  }

273  private void init(int initsize, int maxsize) {
274  // Poolは大したサイズではないので、最初からmaxsize分の配列を作ってしまう

276  pool = new Pool[maxsize];
277  for(int i = 0; i < pool.Length; i++) {
278  pool[i] = new Pool(i);
279  }
280  connectCount = 0;
281  maxConnectCount = 0;
282  }

284  private void purge(bool force) {
285  int count = 0; // 接続中のカウントをついでに行なう
286  foreach(Pool p in pool) {
287  if(p.busy) {
288  count++;
289  } else if((p.dbcon != null) && (force
290  || (p.timer.ElapsedMilliseconds >= ReuseTimeout*1000)
291  || (p.dbcon.DBConnection == null)
292  || (p.dbcon.DBConnection.State != ConnectionState.Open))) {
293  p.dbcon.Dispose();
294  p.dbcon = null;
295  p.busy = false;
296  }
297  }
298  connectCount = count;
299  if(maxConnectCount < count)
300  maxConnectCount = count;
301  }

303  private void dumpStatus(OpeLog logger, TextWriter writer) {
304  int count = 0;
305  for(int i = 0; i < pool.Length; i++) {
306  Pool p = pool[i];
307  string stat = null;
308  if(p.busy) {
309  if(p.dbcon == null)
310  stat = "corrupted (busy but disposed)";
311  else if(p.dbcon.DBConnection == null)
312  stat = "corrupted (busy but no DB connection)";
313  else if(p.dbcon.DBConnection.State != ConnectionState.Open)
314  stat = "corrupted (busy but connection is not open)";
315  else
316  stat = "busy, SQL='"+(p.dbcon.CurrentSQL??"")+"'";
317  } else if(p.dbcon != null) {
318  if(p.timer.ElapsedMilliseconds >= ReuseTimeout*1000)
319  stat = String.Format("expired ({0}msec), waiting for dispose", p.timer.ElapsedMilliseconds);
320  else if(p.dbcon.DBConnection == null)
321  stat = "connection closed, waiting for dispose";
322  else if(p.dbcon.DBConnection.State != ConnectionState.Open)
323  stat = "invalid connection state "+p.dbcon.DBConnection.State.ToString();
324  else
325  stat = String.Format("free ({0}msec)", p.timer.ElapsedMilliseconds);
326  }
327  if(stat != null) {
328  string title = String.Format("DBConPool[{0}:{1}]#{2:D4}", this.Server, this.DBName, i);
329  if(logger != null)
330  logger.Log(title, OpeLog.Level.DEBUG, stat);
331  if(writer != null)
332  writer.WriteLine(title+": "+stat);
333  count++;
334  }
335  }
336  if(count == 0) {
337  string title = String.Format("DBConPool[{0}:{1}]", this.Server, this.DBName);
338  string msg = "No alive connection pools.";
339  if(logger != null)
340  logger.Log(title, OpeLog.Level.DEBUG, msg);
341  if(writer != null)
342  writer.WriteLine(title+": "+msg);
343  }
344  }
