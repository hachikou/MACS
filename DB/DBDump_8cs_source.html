78  /// <summary>
79  /// バックアップファイルを指定ストリームに出力する
80  /// </summary>
81  /// <param name="sw">出力ストリーム</param>
82  /// <param name="db">DB接続</param>
83  /// <param name="password">ZIPパスワード</param>
84  /// <param name="aesEncryption">AES符号化を用いるかどうか</param>
85  public void Backup(Stream sw, DBCon db, string password=null, bool aesEncryption=false) {
86  if(rnd == null)
87  rnd = new Random();
88  string postfix = "."+rnd.Next().ToString();
89  try {
90  timer.Restart();
91  // まず一時ファイルにCSVダンプアウトする
92  int tblcount = 0;
93  foreach(string tblname in tables) {
94  tblcount++;
95  progress("dumping {0} ({1}/{2})", tblname, tblcount, tables.Count);
96  string tmpfilepath = Path.Combine(TempDir, tblname+postfix);
97  DBTable tbl = new DBTable(db, tblname);
98  tbl.DumpCSV(tmpfilepath);
99  }
100  // それをZIPでまとめる
101  using(ZipFile zip = new ZipFile()) {
102  if(!String.IsNullOrEmpty(password)) {
103  zip.Password = password;
104  if(aesEncryption)
105  zip.Encryption = EncryptionAlgorithm.WinZipAes256;
106  }
107  tblcount = 0;
108  foreach(string tblname in tables) {
109  tblcount++;
110  progress("compressing {0} ({1}/{2})", tblname, tblcount, tables.Count);
111  string tmpfilepath = Path.Combine(TempDir, tblname+postfix);
112  zip.AddFile(tmpfilepath).FileName = tblname;
113  }
114  progress("start saving");
115  zip.Save(sw);
116  progress("finished saving");
117  }
118  } finally {
119  foreach(string tblname in tables) {
120  FileInfo tmpfile = new FileInfo(Path.Combine(TempDir, tblname+postfix));
121  if(tmpfile.Exists)
122  tmpfile.Delete();
123  }
124  }

126  }

136  /// <para>
137  /// ファイルフォーマットが適切であるかがチェックされます。
138  /// 各レコードのカラム数が適切であるかどうかはチェックされますが、データ

142  /// にはなりません。
143  /// ただし、ファイル内に必要なテーブルが一つもない場合には、エラーになり
144  /// ます。
145  /// </para>
146  /// </remarks>

148  db.LOG_DEBUG("Checking backup file {0}", filepath);
149  using(Stream sr = FileUtil.BinaryReader(filepath)) {
150  if(sr == null)
151  throw new IOException(String.Format("Can't open {0} for reading.", filepath));
152  return Check(sr, db, password, aesEncryption);
153  }

202  /// バックアップファイルから復元する
203  /// </summary>
204  /// <param name="sr">入力ストリーム</param>
205  /// <param name="db">DB接続</param>
206  /// <param name="password">ZIPパスワード</param>
207  /// <param name="aesEncryption">AES符号化を用いるかどうか</param>
208  /// <param name="truncateFlag">テーブルを全てTruncateするかどうか</param>
209  /// <returns>復元したテーブル数</returns>
210  public int Load(Stream sr, DBCon db, string password=null, bool aesEncryption=false, bool truncateFlag=false) {
211  Stopwatch timer = new Stopwatch();
212  bool oldUseDebugLog = db.UseDebugLog;
213  bool oldUseConsoleLog = db.UseConsoleLog;
214  db.UseDebugLog = false;
215  db.UseConsoleLog = false;
216  try {
217  timer.Start();
218  if(truncateFlag) {
219  foreach(string x in tables) {
220  progress("truncating {0}", x);
221  DBTable tbl = new DBTable(db, x);
222  tbl.Truncate();
223  }
224  }
225  return load(sr, db, password, aesEncryption, false, !truncateFlag);
226  // ここでtruncateしているので、この先ではtruncateの必要が無いことに注意
227  } finally {
228  db.UseDebugLog = oldUseDebugLog;
229  db.UseConsoleLog = oldUseConsoleLog;
230  }
231  }

233  private int load(Stream sr, DBCon db, string password, bool aesEncryption, bool testRun, bool truncateFlag) {
234  DirectoryInfo dir = null;
235  try {
236  dir = extractFiles(sr, password, aesEncryption);
237  FileInfo[] files = dir.GetFiles();
238  // ファイル内容チェック
239  int fileCount = 0;
240  int count = 0;
241  foreach(FileInfo fi in files) {
242  fileCount++;
243  if(!tables.Contains(fi.Name)) {
244  progress("found {0} ({1}/{2}), skip it", fi.Name, fileCount, files.Length);
245  continue;
246  }
247  DBTable tbl = new DBTable(db, fi.Name);
248  if(!testRun && truncateFlag) {
249  progress("truncationg {0}", fi.Name);
250  tbl.Truncate();
251  }
252  if(testRun)
253  progress("checking {0} ({1}/{2})", fi.Name, fileCount, files.Length);
254  else
255  progress("loading {0} ({1}/{2})", fi.Name, fileCount, files.Length);
256  int recs = tbl.LoadCSV(fi.FullName, !testRun, testRun);
257  progress("table {0} has {1} records.", fi.Name, recs);
258  count++;
259  }
260  return count;
261  } finally {
262  if((dir != null) && dir.Exists)
263  dir.Delete(true);
264  }
265  }

267  private DirectoryInfo extractFiles(Stream sr, string password, bool aesEncryption) {
268  // 一時ディレクトリ作成
269  if(rnd == null)
270  rnd = new Random();
271  DirectoryInfo dir = new DirectoryInfo(Path.Combine(TempDir, "dbdump"+rnd.Next().ToString()));
272  dir.Create();
273  try {
274  // ファイル展開
275  using(ZipFile zip = ZipFile.Read(sr)) {
276  foreach(ZipEntry e in zip) {
277  if(!String.IsNullOrEmpty(password)){
278  e.Password = password;
279  if(aesEncryption)
280  e.Encryption = EncryptionAlgorithm.WinZipAes256;
281  }
282  if(tables.Contains(e.FileName))
283  e.Extract(dir.FullName);
284  }
285  }
286  } catch(Exception ex) {
287  if(dir.Exists)
288  dir.Delete(true);
289  throw ex;
290  }
291  return dir;
292  }
