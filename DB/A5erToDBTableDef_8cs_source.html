27  /// <summary>
28  /// コンバートする
29  /// </summary>
30  public int Convert(string a5erfile, string outdir) {
31  if(!Directory.Exists(outdir))
32  Directory.CreateDirectory(outdir);
33  int count = 0;
34  using(StreamReader sr = FileUtil.Reader(a5erfile, Encoding.UTF8)) {
35  if(sr == null)
36  throw new IOException("Can't open "+a5erfile+" for reading");
37  Regex pat_type = new Regex(@"^[\*@]?(\w+)(\(([0-9,]+)\))?$");
38  DBTableDef tabledef = null;
39  string line;
40  while((line = sr.ReadLine()) != null) {
41  line = line.Trim();
42  if(line.StartsWith("[") && line.EndsWith("]")) {
43  if((tabledef != null) && (tabledef.Columns != null)) {
44  tabledef.SaveAs(Path.Combine(outdir, tabledef.XmlFileName));
45  count++;
46  }
47  tabledef = null;
48  }
49  if(line == "[Entity]") {
50  tabledef = new DBTableDef();
51  continue;
52  }
53  if(tabledef == null)
54  continue;
55  string[] kv = line.Split("=".ToCharArray(), 2);
56  if(kv.Length < 2)
57  continue;
58  switch(kv[0]) {
59  case "PName":
60  tabledef.Id = kv[1];
61  tabledef.Name = kv[1];
62  break;
63  case "LName":
64  tabledef.Expr = kv[1];
65  break;
66  case "Comment":
67  tabledef.Note = kv[1].Replace(@"\n","\n");
68  break;
69  case "Field":
70  {
71  string[] x = StringUtil.SplitCSV(kv[1]);
72  if(x.Length < 7)
73  throw new Exception("Invalid Field declaration.");
74  Match mo = pat_type.Match(x[2]);
75  if(!mo.Success)
76  throw new Exception("Invalid type declaration");
77  string type = mo.Groups[1].Value;
78  int length = 0;
79  int fractional = 0;
80  if(mo.Groups[2].Success) {
81  string[] xx = mo.Groups[3].Value.Split(",".ToCharArray());
82  if(xx.Length < 2) {
83  length = StringUtil.ToInt(xx[0]);
84  fractional = 0;
85  } else {
86  length = StringUtil.ToInt(xx[0]);
87  fractional = StringUtil.ToInt(xx[1]);
88  }
89  }
90  DBColumnDef coldef = new DBColumnDef(x[1], // Name
91  type, length, fractional,
92  x[3]=="", // Nullable
93  x[4]!="", // PK
94  x[5]);
95  coldef.Expr = x[0];
96  coldef.Note = x[6].Replace(@"\n","\n");
97  tabledef.AddColumn(coldef);
98  }
99  break;
100  case "Index":
101  {
102  kv = kv[1].Split("=".ToCharArray(), 2);
103  if(kv.Length < 2)
104  throw new Exception("Invalid Index declaration.");
105  string[] x = StringUtil.SplitCSV(kv[1]);
106  if(x.Length < 2)
107  throw new Exception("Invalid Index declaration..");
108  string[] columns = new string[x.Length-1];
109  for(int i = 0; i < x.Length-1; i++)
110  columns[i] = x[i+1];
111  DBIndexDef inddef = new DBIndexDef(columns, x[0]=="1");
112  tabledef.AddIndex(inddef);
113  }
114  break;
115  }
116  }
117  if((tabledef != null) && (tabledef.Columns != null)) {
118  tabledef.SaveAs(Path.Combine(outdir, tabledef.XmlFileName));
119  count++;
120  }
121  sr.Close();
122  }
123  return count;
124  }
