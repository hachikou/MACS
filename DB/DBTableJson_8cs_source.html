9 /// <summary>
10 /// DBTableとNDJsonを結びつける拡張メソッド
11 /// </summary>
12 public static class DBTableJson {
13 
14  /// <summary>
15  /// DBTableのQuery結果をNDJsonで取り出す。
16  /// </summary>
17  /// <remarks>
18  /// <para>
19  /// 得られるJSONは、{"columns":["カラム1","カラム2",...],
20  /// "list":[["値1-1","値1-2",...],["値2-1","値2-2",...],...]} という形式になります。
21  /// </para>
22  /// </remarks>
23  /// <param name="tbl">対象DBTable</param>
24  /// <param name="limit">最大取り出し件数。0の場合無制限</param>
25  /// <param name="offset">何件目以降を返すか。先頭レコードは0</param>
26  public static NDJson GetJson(this DBTable tbl, int limit=0, int offset=0) {
27  NDJson json = new NDJson("columns", tbl.Columns);
28  NDJson list = new NDJson();
29  using(DBReader reader = tbl.Query(limit,offset)) {
30  string[] item;
31  while((item = reader.Get()) != null) {
32  list.Add(new NDJson(item));
33  }
34  }
35  json.Set("list",list);
36  return json;
37  }
38 }
