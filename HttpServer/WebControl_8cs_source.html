50  /// <summary>
51  /// CSSクラスを追加する
52  /// </summary>
53  /// <remarks>
54  /// <para>
55  /// すでにそのクラスが設定されている場合には何もしない
56  /// </para>
57  /// </remarks>
58  public void AddCssClass(string classname) {
59  if(String.IsNullOrEmpty(CssClass)) {
60  CssClass = classname;
61  return;
62  }
63  foreach(string i in CssClass.Split(" ".ToCharArray())) {
64  if(i == classname)
65  return;
66  }
67  CssClass = CssClass+" "+classname;
68  }
69 
70  /// <summary>
71  /// CSSクラスを削除する
72  /// </summary>
73  /// <remarks>
74  /// <para>
75  /// そのクラスが設定されていない場合には何もしない
76  /// </para>
77  /// </remarks>
78  public void RemoveCssClass(string classname) {
79  if(String.IsNullOrEmpty(CssClass)) {
80  return;
81  }
82  StringBuilder sb = new StringBuilder();
83  foreach(string i in CssClass.Split(" ".ToCharArray())) {
84  if(i != classname) {
85  if(sb.Length > 0)
86  sb.Append(' ');
87  sb.Append(i);
88  }
89  }
90  CssClass = sb.ToString();
91  }

173  /// <summary>
174  /// Formの値をValueに取り込む
175  /// </summary>
176  /// <remarks>
177  /// <para>
178  /// テンプレートHTML内で Name= を指定している場合、正しい値を読み取る
179  /// ことができません。（Fetchをする時点ではテンプレートをパースしてい
180  /// ないため。）
181  /// </para>
182  /// </remarks>
183  public virtual void Fetch(HttpPage page, object defaultValue=null) {
184  string val = page.Fetch(Name, null);
185  if(val == null)
186  Value = defaultValue;
187  else if(defaultValue is int)
188  Value = StringUtil.ToInt(val);
189  else if(defaultValue is double)
190  Value = StringUtil.ToDouble(val);
191  else if(defaultValue is bool)
192  Value = StringUtil.ToBool(val);
193  else
194  Value = val;
195  }

198  /// <summary>
199  /// タグ内の共通オプションをレンダリングする
200  /// </summary>

203  // Valueはレンダリングされない事に注意。

212  if(ID != null){
213  sb.Append(" id=\"");
214  sb.Append(HE(ID));
215  sb.Append("\"");
216  }
217  if(CssClass != null){
218  sb.Append(" class=\"");
219  sb.Append(CssClass);
220  sb.Append("\"");
221  }
222  if(Disabled)
223  sb.Append(" disabled=\"disabled\"");
224  if(Width != null){
225  sb.Append(" style=\"width:");
226  sb.Append(Width.ToString());
227  if(typeof(int).IsInstanceOfType(Width))
228  sb.Append("px");
229  sb.Append(";\"");
230  }
231  if(OnClick != null){
232  sb.Append(" onclick=\"");
233  sb.Append(OnClick);
234  sb.Append("\"");
235  }
236  }
