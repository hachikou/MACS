95  foreach(T x in list) {

99  if(x.Equals(xx)) {
100  ex = true;

102  }
103  }
104  }
105  if(ex)
106  continue;
107  string val = x.ToString();

109  sb.Append("<option id=\"");

111  sb.Append("\" value=\"");

115  sb.Append(" selected=\"selected\"");
116  sb.Append(">");
117  string vv;
118  if((Text == null) || !Text.TryGetValue(x, out vv))
119  vv = _(typeof(T).Name+"."+val);
120  sb.Append(HE(vv));
121  sb.Append("</option>");
122  }
123  sb.Append("</select>");
124  RenderInLineError(sb);

126  }
