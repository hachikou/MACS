231  /// <summary>

240  protected void Render(string html) {
241  HttpListenerResponse response = m_context.Response;
242  response.ContentEncoding = m_encoding;
243  if(m_encoding == Encoding.UTF8)
244  response.ContentType = "text/html; charset=utf-8";
245  else
246  response.ContentType = "text/html";
247  byte[] buffer = m_encoding.GetBytes(html);
248  response.ContentLength64 = buffer.Length;
249  response.OutputStream.Write(buffer, 0, buffer.Length);
250  response.OutputStream.Close();
251  m_rendered = true;
252  }

443  using(FileStream fs = FileUtil.BinaryReader(realfilename)) {

450  /// <summary>
451  /// クライアントにファイルを送信する
452  /// </summary>
453  /// <param name="fs">送信するファイルのファイルストリーム</param>
454  /// <param name="filename">クライアントに通知するファイル名</param>
455  /// <param name="mime">MIMEタイプ名(Content-Type)</param>
456  protected void SendFile(FileStream fs, string filename, string mime) {
457  if(fs == null)
458  throw new ArgumentNullException("No file stream to send.");
459  if(m_rendered)
460  throw new InvalidOperationException("Content is already rendered.");
461  HttpListenerResponse response = m_context.Response;
462  if(!String.IsNullOrEmpty(mime))
463  response.ContentType = mime;
464  else
465  response.ContentType = "application/octet-stream";
466  string disposition = "attachment";
467  if(!String.IsNullOrEmpty(filename))
468  disposition += "; filename=\""+filename+"\"";
469  response.AddHeader("Content-disposition", disposition);
470  response.ContentLength64 = fs.Length;
471  byte[] buf = new byte[4096];
472  int len;
473  while((len = fs.Read(buf, 0, buf.Length)) > 0)
474  response.OutputStream.Write(buf, 0, len);
475  response.OutputStream.Close();
476  m_rendered = true;
477  }

715  private static void AppendQueryString(NameValueCollection dict, string qstr) {
716  foreach(string str in qstr.Split("&;".ToCharArray())) {
717  string[] kv = str.Split("=".ToCharArray(),2);
718  if(kv.Length == 1)
719  dict.Add(Uri.UnescapeDataString(kv[0].Replace('+',' ')),"");
720  else if(kv.Length == 2)
721  dict.Add(Uri.UnescapeDataString(kv[0].Replace('+',' ')), Uri.UnescapeDataString(kv[1].Replace('+', ' ')));
722  }

725  private void FetchMultipartForm(HttpListenerRequest req) {
726  #if FORMDEBUG
727  Console.WriteLine("Entering FetchMultipartForm");
728  #endif
729  // まずboundary文字列を獲得する
730  string boundary = null;
731  foreach(string i in req.ContentType.Split(";".ToCharArray())){
732  string[] kv = i.Split("=".ToCharArray(), 2);
733  if(kv.Length != 2)
734  continue;
735  if(kv[0].Trim().ToLower() == "boundary")
736  boundary = kv[1].Trim().Trim("\"".ToCharArray());
737  }
738  if((boundary == null) || (boundary == "")){
739  LOG_ERR("No boundary for multipart/form-data");
740  return;
741  }
742  #if FORMDEBUG
743  Console.WriteLine("boundary="+boundary);
744  #endif
745  boundary = "--"+boundary;
746  string finalboundary = boundary+"--";
747 
748  //Encoding enc = req.ContentEncoding; //クライアントは正しくContentEncodingを渡さない。
749  Encoding enc = m_encoding;
750  Decoder dec = enc.GetDecoder();
751  byte[] buf = new byte[512];
752  int buflen = 0;
753  bool inhead = false;
754  string name = null;
755  StringBuilder tmpvalue = null;
756  HttpPostedFile tmpfile = null;
757  #if FORMDEBUG
758  int n = 0;
759  #endif
760  int ch;
761  try {
762  // 入力をスキャン
763  using(Stream sr = new BufferedStream(req.InputStream)) {
764  while((ch = sr.ReadByte()) >= 0){
765  #if FORMDEBUG
766  if(n >= 16){
767  Console.WriteLine();
768  n = 0;
769  }
770  Console.Write("[{0:X2}]", ch);
771  if((ch >= 0x20) && (ch <= 0xfe))
772  Console.Write(Char.ConvertFromUtf32(ch));
773  else
774  Console.Write(' ');
775  n++;
776  #endif
777 
778  // ラインバッファに溜め込む
779  buf[buflen++] = (byte)ch;
780  if(buflen >= buf.Length){
781  // ラインバッファがいっぱい。
782  if(tmpvalue != null){
783  // 文字列受信中は部分デコードする。
784  char[] tmpchars = new char[buflen];
785  int tmpcharslen = dec.GetChars(buf, 0, buflen, tmpchars, 0, false);
786  tmpvalue.Append(tmpchars, 0, tmpcharslen);
787  #if FORMDEBUG
788  Console.WriteLine();
789  Console.WriteLine("Append {0}chars", tmpcharslen);
790  #endif
791  // TODO: あまりにも巨大なデータを受けた場合は入力スキャンを途中で止める事
792  } else if(tmpfile != null){
793  // ファイル受信中は書き出す。
794  tmpfile.m_stream.Write(buf, 0, buflen);
795  #if FORMDEBUG
796  Console.WriteLine();
797  Console.WriteLine("Write {0}bytes to {1}", buflen, tmpfile.m_innerfilename);
798  #endif
799  // TODO: あまりにも巨大なファイルを受けた場合は入力スキャンを途中で止める事
800  }
801  buflen = 0;
802  continue;
803  }
804 
805  if(ch == 0x0a){ // ライン終端
806  try {
807  char[] tmpchars = new char[buflen];
808  int tmpcharslen;
809  if((buflen > 1) && (buf[buflen-2] == 0x0d)) // LFの前にCRがあってもなくてもよいように。
810  tmpcharslen = dec.GetChars(buf, 0, buflen-2, tmpchars, 0, true);
811  else
812  tmpcharslen = dec.GetChars(buf, 0, buflen-1, tmpchars, 0, true);
813  dec = enc.GetDecoder(); // デコーダをリセットしておく
814  string line = new string(tmpchars, 0, tmpcharslen);
815  #if FORMDEBUG
816  Console.WriteLine();
817  Console.WriteLine("Line '"+line+"'");
818  n = 0;
819  #endif
820  if(inhead){
821  // パートヘッダ中
822  if(line == ""){
823  #if FORMDEBUG
824  Console.WriteLine("Begin part body");
825  #endif
826  inhead = false;
827  }else if(line.StartsWith("content-disposition:", true, CultureInfo.InvariantCulture)){
828  // 属性値取り出し
829  foreach(string i in line.Split(";".ToCharArray())){
830  string[] kv = i.Split("=".ToCharArray(), 2);
831  if(kv.Length != 2)
832  continue;
833  string k = kv[0].Trim().ToLower();
834  switch(k){
835  case "name":
836  name = kv[1].Trim().Trim("\"".ToCharArray());
837  break;
838  case "filename":
839  tmpfile = new HttpPostedFile();
840  tmpfile.m_filename = kv[1].Trim().Trim("\"".ToCharArray());
841  if(!Directory.Exists(m_server.TemporaryDirectory))
842  Directory.CreateDirectory(m_server.TemporaryDirectory);
843  tmpfile.m_innerfilename = Path.Combine(m_server.TemporaryDirectory, TemporaryPrefix+m_rand.Next(0x7fffffff).ToString("X8"));
844  tmpfile.m_stream = FileUtil.BinaryWriter(tmpfile.m_innerfilename);
845  if(tmpfile.m_stream == null){
846  LOG_ERR(string.Format("Can't create '{0}'", tmpfile.m_innerfilename));
847  tmpfile = null;
848  }
849  break;
850  }
851  }
852  if(tmpfile == null)
853  tmpvalue = new StringBuilder();
854  #if FORMDEBUG
855  Console.WriteLine("Fetch '{0}' part", name);
856  if(tmpfile != null)
857  Console.WriteLine("Inner filename={0}", tmpfile.m_innerfilename);
858  #endif
859  }
860  buflen = 0;
861  } else if(tmpvalue != null) {
862  // フォーム要素値獲得中
863  if((line == boundary) || (line == finalboundary)){
864  if(tmpvalue.Length > 0)
865  tmpvalue.Length--;
866  if(name == null)
867  name = "";
868  m_form.Add(name, tmpvalue.ToString());
869  #if FORMDEBUG
870  Console.WriteLine("{0}='{1}'", name, tmpvalue.ToString());
871  #endif
872  tmpvalue = null;
873  buflen = 0;
874  inhead = true;
875  }else{
876  tmpvalue.Append(line);
877  tmpvalue.Append('\n');
878  // TODO: あまりにも巨大なデータを受けた場合は入力スキャンを途中で止める事
879  }
880  } else if(tmpfile != null){
881  // ファイル受信中
882  if((line == boundary) || (line == finalboundary)){
883  if(tmpfile.m_stream.Length >= 2)
884  tmpfile.m_stream.SetLength(tmpfile.m_stream.Length-2);
885  tmpfile.m_stream.Close();
886  tmpfile.m_stream = null;
887  if(name == null)
888  name = "";
889  m_fileform[name] = tmpfile;
890  #if FORMDEBUG
891  Console.WriteLine("{0}='{1}', {2}bytes", name, tmpfile.FileName, new FileInfo(tmpfile.m_innerfilename).Length);
892  #endif
893  tmpfile = null;
894  buflen = 0;
895  inhead = true;
896  }
897  } else {
898  if((line == boundary) || (line == finalboundary)){
899  #if FORMDEBUG
900  Console.WriteLine("Got boundary");
901  #endif
902  buflen = 0;
903  inhead = true;
904  }
905  }
906  } catch(ArgumentException e){
907  #if FORMDEBUG
908  Console.WriteLine(e.Message);
909  #endif
910  e.GetType();
911  // just ignore
912  }
913  // ファイル受信中は書き出す
914  if((tmpfile != null) && (buflen > 0)){
915  #if FORMDEBUG
916  Console.WriteLine("Write {0}bytes to {1}", buflen, tmpfile.m_innerfilename);
917  #endif
918  tmpfile.m_stream.Write(buf, 0, buflen);
919  // TODO: あまりにも巨大なデータを受けた場合は入力スキャンを途中で止める事
920  }
921  buflen = 0;
922  }
923  }
924  }
925 
926  #if FORMDEBUG
927  Console.WriteLine();
928  #endif
929  } finally {
930  #if FORMDEBUG
931  Console.WriteLine("Leaving FetchMultipartForm");
932  #endif
933  if(tmpvalue != null)
934  tmpvalue = null;
935  if(tmpfile != null){
936  tmpfile.Dispose();
937  tmpfile = null;
938  }
939  }
