13 /// <summary>

18  /// <summary>
19  /// ファイル名
20  /// </summary>
21  public string FileName {
22  get { return m_filename; }
23  }
24 
25  /// <summary>
26  /// 入力ストリーム
27  /// </summary>
28  public Stream InputStream {
29  get {
30  if(m_stream == null){
31  if(string.IsNullOrEmpty(m_innerfilename) || !File.Exists(m_innerfilename))
32  return null;
33  m_stream = FileUtil.BinaryReader(m_innerfilename);
34  }
35  return m_stream;
36  }
37  }
38 
39  /// <summary>
40  /// ファイル長
41  /// </summary>
42  /// <remarks>
43  /// <para>
44  /// SaveAs(string) を行なった後は、本プロパティは0を返す。
45  /// その保存ファイルのサイズを確認する事。
46  /// </para>
47  /// </remarks>
48  public long ContentLength {
49  get {
50  if(string.IsNullOrEmpty(m_innerfilename) || !File.Exists(m_innerfilename))
51  return 0;
52  FileInfo fi = new FileInfo(m_innerfilename);
53  return fi.Length;
54  }
55  }
56 
57  internal string m_filename;
58  internal string m_innerfilename;
59  internal FileStream m_stream;

62  /// デフォルトコンストラクタ。

65  }

74  /// <summary>
75  /// ディスポーザ
76  /// </summary>
77  public void Dispose() {
78  if(m_stream != null){
79  m_stream.Close();
80  m_stream = null;
81  }
82  if(!string.IsNullOrEmpty(m_innerfilename) && File.Exists(m_innerfilename)){
83  File.Delete(m_innerfilename);
84  }
85  m_innerfilename = null;
86  }
87 
88  /// <summary>
89  /// 受信ファイルを名前を付けて保存する。
90  /// </summary>
91  /// <remarks>
92  /// <para>
93  /// 本メソッド呼び出し以降はInputStreamが使えなくなる事に注意。
94  /// </para>
95  /// </remarks>
96  public void SaveAs(string path) {
97  if(string.IsNullOrEmpty(m_innerfilename) || !File.Exists(m_innerfilename))
98  throw new FileNotFoundException("HttpPostedFile does not have inner file.");
99  if(m_stream != null){
100  m_stream.Close();
101  m_stream = null;
102  }
103  File.Delete(path);
104  File.Move(m_innerfilename, path);
105  }
