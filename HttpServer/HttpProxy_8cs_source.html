76  /// <summary>
77  /// HTTPのプロキシを行なう
78  /// </summary>
79  public bool Proxy(HttpListenerRequest request, HttpListenerResponse response, string url=null) {
80  Uri orgUrl = request.Url;
81  if(url == null) {
82  url = orgUrl.PathAndQuery;
83  }
84  string remoteUrl = String.Format("http://{0}:{1}{2}", ServerName, PortNo, rewriteUrl(url, sourceUrlPrefix, remoteUrlPrefix));
85  LOG_DEBUG("Remote URL={0}", remoteUrl);
86  try{
87  HttpWebRequest remoteRequest = WebRequest.CreateHttp(remoteUrl);
88  foreach(string key in request.Headers.Keys) {
89  try {
90  remoteRequest.Headers[key] = request.Headers[key];
91  DETAIL("Set request header {0}={1}", key, request.Headers[key]);
92  } catch(ArgumentException) {
93  DETAIL("Failed to set request header {0}={1}", key, request.Headers[key]);
94  }
95  }
96  remoteRequest.Timeout = Timeout;
97  remoteRequest.AllowAutoRedirect = false;
98  if(FullDebug) {
99  foreach(Cookie cookie in request.Cookies) {
100  DETAIL("Request cookie '{0}'", cookie.ToString());
101  }
102  }
103  remoteRequest.Method = request.HttpMethod.ToUpper();
104  if(FullDebug) {
105  foreach(string key in remoteRequest.Headers.Keys) {
106  DETAIL("Request header: {0}={1}", key, remoteRequest.Headers[key]);
107  }
108  }
109  if(request.UrlReferrer != null && !String.IsNullOrEmpty(request.UrlReferrer.AbsoluteUri))
110  remoteRequest.Referer = request.UrlReferrer.AbsoluteUri;
111  remoteRequest.UserAgent = request.UserAgent;
112 
113  byte[] buf = new byte[1024*4];
114  int len, totallen;
115 
116  // POSTデータを転送する
117  if(remoteRequest.Method == "POST") {
118  //remoteRequest.ContentLength = request.ContentLength64;
119  remoteRequest.ContentType = request.ContentType;
120  using(Stream remoteStreamIn = remoteRequest.GetRequestStream()) {
121  totallen = 0;
122  while((len = request.InputStream.Read(buf, 0, buf.Length)) > 0) {
123  remoteStreamIn.Write(buf, 0, len);
124  totallen += len;
125  }
126  remoteStreamIn.Close();
127  DETAIL("Written {0}bytes to request stream", totallen);
128  }
129  }
130 
131  // 応答を受信してクライアントに返す
132  using(HttpWebResponse remoteResponse = (HttpWebResponse)remoteRequest.GetResponse()) {
133  string[] setCookieList = null;
134  string[] delimiter = {","};
135  response.Cookies = remoteResponse.Cookies;
136  foreach(string key in remoteResponse.Headers.Keys) {
137  try {
138  if (key == "Set-Cookie") {
139  setCookieList = remoteResponse.Headers[key].Replace(", ","[HTTPPROXY_COMMA_SPACE]").Split(delimiter, StringSplitOptions.RemoveEmptyEntries);
140  } else {
141  response.Headers[key] = remoteResponse.Headers[key];
142  }
143  DETAIL("Set response header {0}={1}", key, remoteResponse.Headers[key]);
144  } catch(ArgumentException) {
145  DETAIL("Failed to set response header {0}={1}", key, remoteResponse.Headers[key]);
146  }
147  }
148  if (setCookieList != null) {
149  foreach(string cookie in setCookieList) {
150  response.AppendHeader("Set-Cookie", cookie.Replace("[HTTPPROXY_COMMA_SPACE]", ", "));
151  }
152  }
153  response.ProtocolVersion = remoteResponse.ProtocolVersion;
154  response.StatusCode = (int)remoteResponse.StatusCode;
155  if(isRedirect(response.StatusCode)) {
156  string location = remoteResponse.Headers.Get("Location");
157  if(location == null) {
158  LOG_NOTICE("{0} returned Http-Status#{1}, but no Location", remoteUrl, response.StatusCode);
159  } else {
160  LOG_DEBUG("Redirect to {0}", location);
161  string redirectPath;
162  try {
163  Uri redirectUrl = new Uri(location);
164  redirectPath = redirectUrl.PathAndQuery;
165  } catch(UriFormatException) {
166  redirectPath = location;
167  }
168  response.RedirectLocation = rewriteUrl(redirectPath, remoteUrlPrefix, sourceUrlPrefix);
169  }
170  }
171  response.StatusDescription = remoteResponse.StatusDescription;
172  response.KeepAlive = false;
173  if(FullDebug) {
174  foreach(string key in response.Headers.Keys) {
175  DETAIL("Response header: {0}={1}", key, response.Headers[key]);
176  }
177  }
178  Stream remoteStream = remoteResponse.GetResponseStream();
179  buf = new byte[1024*4];
180  totallen = 0;
181  while((len = remoteStream.Read(buf, 0, buf.Length)) > 0) {
182  response.OutputStream.Write(buf, 0, len);
183  totallen += len;
184  }
185  DETAIL("Written {0}bytes to response stream", totallen);
186  }
187  } catch(ProtocolViolationException ex) {
188  LOG_ERR("Proxy error: {0}", ex.Message);
189  response.StatusCode = 500; // Internal Server Error
190  response.StatusDescription = ex.Message;
191  return false;
192  } catch(WebException ex) {
193  LOG_NOTICE("WebException: {0}: {1}", ex.Status.ToString(), ex.Message);
194  if(ex.Status == WebExceptionStatus.ProtocolError) {
195  response.StatusCode = (int)((HttpWebResponse)ex.Response).StatusCode;
196  response.StatusDescription = ((HttpWebResponse)ex.Response).StatusDescription;
197  } else {
198  response.StatusCode = 500; // Internal Server Error
199  response.StatusDescription = ex.Message;
200  }
201  return true;
202  }
203  return true;
204  }

206  /// <summary>
207  /// WEBサーバに接続できるかどうか確認する
208  /// </summary>
209  public bool CheckConnection() {
210  bool ok = false;
211  string remoteUrl = String.Format("http://{0}:{1}{2}", ServerName, PortNo, remoteUrlPrefix);
212  DETAIL("Check access to {0}", remoteUrl);
213  try {
214  HttpWebRequest remoteRequest = WebRequest.CreateHttp(remoteUrl);
215  remoteRequest.KeepAlive = false;
216  remoteRequest.Method = "HEAD";
217  remoteRequest.Timeout = Timeout;
218  using(HttpWebResponse remoteResponse = (HttpWebResponse)remoteRequest.GetResponse()) {
219  int status = (int)remoteResponse.StatusCode;
220  DETAIL("Got {0} {1}", status, remoteResponse.StatusDescription);
221  if((status >= 100) && (status < 500))
222  ok = true; // とりあえずサーバエラー以外のなんか応答があればOK
223  }
224  } catch(ProtocolViolationException ex) {
225  DETAIL("Connection error: {0}", ex.Message);
226  ok = false;
227  } catch(WebException ex) {
228  DETAIL("WebException: {0}: {1}", ex.Status.ToString(), ex.Message);
229  ok = false;
230  } catch(Exception ex) {
231  LOG_ERR("Fatal error while checking connection: {0}: {1}", ex.GetType().Name, ex.Message);
232  ok = false;
233  }
234  return ok;
235  }

240  private static string cleanUrlPrefix(string urlPrefix) {
241  if(String.IsNullOrEmpty(urlPrefix))
242  return "/";
243  if(!urlPrefix.StartsWith("/"))
244  urlPrefix = "/"+urlPrefix;
245  if(!urlPrefix.EndsWith("/"))
246  urlPrefix = urlPrefix+"/";
247  return urlPrefix;
248  }
249 
250  private static string rewriteUrl(string url, string srcPrefix, string dstPrefix) {
251  if(url == null)
252  return null;
253  if(url.StartsWith(srcPrefix))
254  url = dstPrefix+url.Substring(srcPrefix.Length);
255  return url;
256  }

258  private static bool isRedirect(int code) {
259  switch(code){
260  case 301: // Moved Permanently
261  case 302: // Found
262  case 303: // See Other
263  case 307: // Temporary Redirect
264  case 308: // Permanent Redirect
265  return true;
266  default:
267  return false;
268  }
269  }
