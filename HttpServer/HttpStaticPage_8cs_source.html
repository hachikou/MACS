52  /// 次の拡張子のContent-Typeが設定される。

113  /// <summary>
114  /// Request URLに指定されたファイルを出力する。
115  /// </summary>
116  /// <remarks>
117  /// <para>
118  /// 本メソッドはHttpServerが呼び出します。
119  /// </para>
120  /// </remarks>
121  public override void PageLoad(string param) {
122  // ファイル名の正当性の確認
123  string fname = param.Replace('\\','/');
124  if(!fname.StartsWith("/") || fname.Contains("/../")){
125  SetNoCache();
126  Render(403);
127  return;
128  }
129 
130  // ディレクトリが指定されたかどうか
131  if(fname.EndsWith("/") || DirectoryExists(m_dir+fname))
132  fname = Path.Combine(fname,IndexFile);
133 
134  string ext = Path.GetExtension(fname);
135  fname = GetNlsFileName(m_dir+fname);
136 
137  if(!FileExists(fname)){
138  SetNoCache();
139  Render(404);
140  return;
141  }
142 
143  // キャッシュコントロール
144  DateTime mtime = GetLastWriteTimeUtc(fname);
145  Response.AddHeader("Last-Modified", mtime.ToString("r"));
146  string etag = "\""+BitConverter.ToString(MD5.Create().ComputeHash(Encoding.UTF8.GetBytes(fname+mtime.ToBinary().ToString())))+"\"";
147  Response.AddHeader("ETag", etag);
148  if(string.Compare(Request.Headers.Get("Cache-Control"), "no-cache", true, CultureInfo.InvariantCulture) != 0){
149  try {
150  string reqetag = Request.Headers.Get("If-None-Match");
151  if(reqetag == null){
152  // ETagが指定されていない場合はIf-Modified-Sinceで確認する。
153  DateTime reqtime = DateTime.ParseExact(Request.Headers.Get("If-Modified-Since"), "r", null);
154  if(mtime <= reqtime){
155  LOG_DEBUG("Cached page is newer");
156  Render(304);
157  return;
158  }
159  }else{
160  if(reqetag == etag){
161  LOG_DEBUG("Cached page is unchanged");
162  Render(304);
163  return;
164  }
165  }
166  } catch(Exception){
167  // Just ignore.
168  }
169  }
170 
171  using(Stream f = OpenFile(fname)){
172  if(f == null){
173  SetNoCache();
174  Render(406);
175  return;
176  }
177  Response.ContentEncoding = m_encoding;
178  if(m_contenttype == null){
179  if(m_ext_type == null)
180  SetDefaultContentTypeList();
181  string ext_ = ext.ToLower();
182  if(m_ext_type.ContainsKey(ext_))
183  Response.ContentType = m_ext_type[ext_];
184  else
185  Response.ContentType = default_contenttype;
186  }else{
187  Response.ContentType = m_contenttype;
188  }
189  Int64 len = f.Length;
190  Response.ContentLength64 = len;
191  byte[] buffer = new byte[4096];
192  while(len > 0){
193  int xlen = f.Read(buffer, 0, buffer.Length);
194  if(xlen <= 0)
195  throw new IOException(string.Format("Read error while reading '{0}'({1} bytes left)", fname, len));
196  Response.OutputStream.Write(buffer, 0, xlen);
197  len -= xlen;
198  }
199  f.Close();
200  Response.OutputStream.Close();
201  }
202  }
