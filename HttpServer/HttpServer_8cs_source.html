386  /// <summary>
387  /// HTTPサーバを動かす。
388  /// </summary>
389  /// <remarks>
390  /// <para>
391  /// Stop()が（他のスレッドで）呼び出されるまで、本メソッドは戻らない。
392  /// </para>
393  /// </remarks>
394  /// <param name="initworkers">初期ワーカースレッド数</param>
395  public void Run(int initworkers) {
396  if(IsRunning){
397  LOG_CRIT("Already running.");
398  return;
399  }
400  m_stoprequest = false;
401  if(initworkers <= 0)
402  initworkers = 1;
403 
404  m_workerlist = new List<WorkerContext>();
405  for(int i = 0; i < initworkers; i++){
406  m_workerlist.Add(new WorkerContext(this, i));
407  }
408 
409  try {
410  m_listener.Start();
411  } catch(Exception e) {
412 
413  // Windows 7 でデバッグ実行しているときにここに陥る場合、
414  // 管理者権限で 'netsh http add urlacl url=http://+:10080/ user=ユーザー名' を実行してください。
415 
416  LOG_CRIT(string.Format("Failed to listen on {0}: {1}", m_url, e.Message));
417  foreach(WorkerContext wc in m_workerlist) {
418  wc.Dispose();
419  }
420  m_workerlist = null;
421  throw e;
422  }
423 
424  LOG_INFO(string.Format("Accepting {0}, {1} workers.", m_url, m_workerlist.Count));
425 
426  m_activepages = 0;
427  while (!m_stoprequest) {
428  // コネクトを待つ。
429  HttpListenerContext context;
430  try{
431  m_listening = true;
432  context = m_listener.GetContext();
433  } catch(Exception e) {
434  // Stop()メソッドによってm_listenerがStopさせられたのならOK
435  if(m_stoprequest)
436  break;
437  throw e;
438  } finally {
439  m_listening = false;
440  }
441  // 接続元を確認する
442  lock(m_allownetaddr) {
443  if(m_allownetaddr.Count > 0) {
444  IPAddress remote = context.Request.RemoteEndPoint.Address;
445  if(!checkRemoteAddress(remote)) {
446  context.Response.Abort();
447  LOG_NOTICE("Reject accress from "+remote.ToString());
448  continue;
449  }
450  }
451  }
452  // 空いているワーカースレッドを探す
453  int widx = -1;
454  lock(m_workerlist) {
455  for(int i = 0; i < m_workerlist.Count; i++){
456  if(m_workerlist[i].Status == WorkerStatus.WAITING){
457  if(widx < 0)
458  widx = i;
459  } else if(m_workerlist[i].Status == WorkerStatus.STOPPED) {
460  // 止まってしまったスレッドを回収する
461  LOG_NOTICE("Disposing worker#{0}", i);
462  m_workerlist[i].Dispose();
463  m_workerlist[i] = new WorkerContext(this, i);
464  if(widx < 0)
465  widx = i;
466  }
467  }
468  if(widx < 0){ // All workers are busy.
469  if(m_workerlist.Count >= m_maxworkers) {
470  context.Response.Abort();
471  LOG_ERR("No more workers (limit={0}). Connection from client is aborted.", m_maxworkers);
472  continue;
473  }
474  // ワーカーを増やして対応する
475  widx = m_workerlist.Count;
476  m_workerlist.Add(new WorkerContext(this, widx));
477  }
478  } // end of lock(m_workerlist)
479  // ワーカースレッドにcontextを渡して、処理させる。
480  m_workerlist[widx].Status = WorkerStatus.CONNECTED;
481  m_workerlist[widx].Context = context;
482  m_workerlist[widx].WorkerEvent.Set();
483  m_pagecount++;
484  }
485  try {
486  m_listener.Stop();
487  } catch(Exception) {
488  // just ignore.
489  }
490 
491  foreach(WorkerContext wc in m_workerlist)
492  wc.WorkerEvent.Set();
493  foreach(WorkerContext wc in m_workerlist) {
494  wc.WorkerThread.Join();
495  wc.Dispose();
496  }
497  m_workerlist = null;
498  m_stoprequest = false;
499 
500  LOG_INFO("Finished accepting {0}", m_url);
501  }

572  /// <summary>
573  /// サーバ稼働状況獲得
574  /// </summary>
575  public Statistics GetStatistics() {
576  Statistics stat = new Statistics();
577  if(m_workerlist == null)
578  return stat;
579 
580  lock(m_workerlist) {
581  stat.Workers = m_workerlist.Count;
582  foreach(WorkerContext wc in m_workerlist) {
583  if(wc == null){
584  stat.Stopped++;
585  continue;
586  }
587  stat.PageCount += wc.Count;
588  stat.TotalTime += wc.TotalElapsed;
589  if(stat.MaxTime < wc.MaxElapsed)
590  stat.MaxTime = wc.MaxElapsed;
591  switch(wc.Status){
592  case WorkerStatus.WAITING:
593  stat.Waiting++;
594  break;
595  case WorkerStatus.CONNECTED:
596  stat.Connected++;
597  break;
598  case WorkerStatus.BUSY:
599  stat.Busy++;
600  break;
601  case WorkerStatus.STOPPED:
602  stat.Stopped++;
603  break;
604  }
605  }
606  }
607  return stat;
608  }

618  /// <summary>
619  /// サーバ稼働状況メッセージを得る
620  /// </summary>
621  public string[] GetDebugMessage(bool detail=false) {
622  List<string> res = new List<string>();
623  Statistics stat = GetStatistics();
624  res.Add(String.Format("Workers={0}; {1} waiting, {2} connected, {3} busy, {4} stopped", stat.Workers, stat.Waiting, stat.Connected, stat.Busy, stat.Stopped));
625  res.Add(String.Format("Total {0} pages; mean {1}msec, max {2}msec", stat.PageCount, stat.MeanTime, stat.MaxTime));
626  if(detail && (stat.Workers > 0)) {
627  if(m_workerlist == null) {
628  res.Add("Corrupted (workerlist is gone away)");
629  } else {
630  lock(m_workerlist) {
631  foreach(WorkerContext wc in m_workerlist) {
632  string title = String.Format("Worker#{0}: ", wc.WorkerNumber);
633  if(wc == null) {
634  res.Add(title+"DISPOSED");
635  continue;
636  }
637  res.Add(String.Format("{0}{1} pages={2}, totalTime={3}msec, maxTime={4}msec", title, wc.Status.ToString(), wc.Count, wc.TotalElapsed, wc.MaxElapsed));
638  }
639  }
640  }
641  }
642  return res.ToArray();
643  }

690  public WorkerContext(HttpServer server, int workerNumber) {
691  server.LOG_INFO("Creating worker#{0}", workerNumber);
692  this.WorkerNumber = workerNumber;
693  this.Status = WorkerStatus.WAITING;
694  this.Server = server;
695  this.Context = null;
696  this.WorkerEvent = new AutoResetEvent(false);

698  this.MaxElapsed = 0;
699  this.Timer = new Stopwatch();

719  private void doWork(){
720  HttpServer sv = this.Server;
721  try {
722  sv.LOG_DEBUG("Start worker#{0}", this.WorkerNumber);
723  while(!sv.m_stoprequest){
724  this.WorkerEvent.WaitOne();
725  if(sv.m_stoprequest)
726  break;
727  this.Status = WorkerStatus.BUSY;
728  sv.OnConnect(this);
729  this.Status = WorkerStatus.WAITING;
730  }
731  sv.LOG_DEBUG("Stop worker#{0}", this.WorkerNumber);
732  } catch(ThreadAbortException) {
733  // OnConnect内でログを吐いているので、ここでは何もしなくてよい
734  } catch(Exception ex) {
735  sv.LOG_ERR("Worker#{0} caught exception {1}", this.WorkerNumber, ex.GetType().Name);
736  sv.LOG_EXCEPTION(ex);
737  } finally {
738  this.Status = WorkerStatus.STOPPED;
739  }
740  }

768  private void OnConnect(WorkerContext wc) {
769  wc.Timer.Reset();
770  wc.Timer.Start();
771  wc.Count++;
772  HttpListenerContext context = wc.Context;
773  string requestmemo = string.Format("{0} {1} from {2}",
774  context.Request.HttpMethod,
775  context.Request.RawUrl,
776  context.Request.RemoteEndPoint);
777  LOG_DEBUG("Requested "+requestmemo);
778 
779  // 古いセッションを破棄
780  PurgeSession();
781 
782  context.Response.KeepAlive = false;
783  context.Response.AddHeader("Server", "MACS HttpServer");
784  string path = context.Request.Url.AbsolutePath;
785  int status = 0;
786  try {
787  // デフォルトページ？
788  if((path == "/") && (m_defaultpage != null)){
789  using(HttpPage page = new HttpPage()){
790  page.SetLogger(this.Logger);
791  page.SetServerContext(this, context, path);
792  page.Redirect(m_defaultpage);
793  status = 200;
794  }
795  }
796 
797  // 静的ページを探す
798  if(status == 0) {
799  foreach(KeyValuePair<string,HttpStaticPage> kv in m_staticpagelist){
800  if(path.StartsWith(kv.Key) && (path.Length > kv.Key.Length) && (path[kv.Key.Length] == '/')){
801  if(context.Request.HttpMethod == "GET"){
802  lock(kv.Value){
803  kv.Value.SetLogger(this.Logger);
804  kv.Value.SetServerContext(this, context, path);
805  kv.Value.PageLoad(path.Substring(kv.Key.Length));
806  }
807  status = 200;
808  }else{
809  status = 405;
810  }
811  break;
812  }
813  }
814  }
815 
816  if(status == 0){
817  // アプリケーションページを探す
818  foreach(KeyValuePair<string,Type> kv in m_pagelist){
819  if(path.StartsWith(kv.Key) && ((path.Length == kv.Key.Length) || (path[kv.Key.Length] == '/'))){
820  // 動的ページにマッチした
821  Type pageClass = kv.Value;
822  // 動的ページは同時にMaxActivePages個しかレンダリングしない
823  lock(m_activepages_mutex) {
824  if(++m_activepages > m_maxactivepages) {
825  // オーバーした時は強制的にEmergencyPageでレンダリングする
826  LOG_NOTICE("Number of active pages exceeds MaxActivePages({0}), use {1}", m_maxactivepages, m_emergencypage.Name);
827  pageClass = m_emergencypage;
828  }
829  }
830  int timeout = 0; // ページレンダリングタイムアウト
831  try {
832  if((context.Request.HttpMethod == "GET") || (context.Request.HttpMethod == "POST")){
833  // セッション変数をクッキー管理する。
834  Cookie sessid = context.Request.Cookies[m_servername+"-SID"];
835  ObjectDictionary sess;
836  lock(m_sessiondict){
837  if((sessid != null) && m_sessiondict.ContainsKey(sessid.Value)){
838  sess = m_sessiondict[sessid.Value];
839  }else{
840  sessid = new Cookie(m_servername+"-SID", m_rand.Next(9999999).ToString()+m_rand.Next(9999999).ToString());
841  sess = new ObjectDictionary();
842  m_sessiondict[sessid.Value] = sess;
843  }
844  m_sessiontime[sessid.Value] = DateTime.Now;
845  }
846  sessid.Version = 1;
847  sessid.Path = "/";
848  if(UseCookieExpires && (SessionTimeout > 0)) {
849  sessid.Expires = DateTime.Now.AddSeconds(SessionTimeout);
850  //context.Response.AppendCookie(sessid);
851  // なぜか正しいSet-Cookieヘッダを生成してくれないので、自力でヘッダを作る。
852  context.Response.AppendHeader("Set-Cookie", string.Format("{0}={1}; expires={2}; path={3}", sessid.Name, sessid.Value, sessid.Expires.ToString("r"), sessid.Path));
853  } else {
854  context.Response.AppendHeader("Set-Cookie", string.Format("{0}={1}; path={2}", sessid.Name, sessid.Value, sessid.Path));
855  }
856 
857  // ページレンダラの呼び出し
858  using(HttpPage page = (HttpPage)(pageClass.GetConstructor(Type.EmptyTypes).Invoke(null))){
859  page.SetLogger(this.Logger);
860  page.SetServerContext(this, context, kv.Key.Substring(1));
861  // ページレンダリングタイムアウトの設定
862  timeout = page.Timeout;
863  if(timeout < 0)
864  timeout = DefaultTimeout;
865  if(timeout > 0)
866  NThread.Watchdog(timeout, NThread.WatchdogMode.Abort);
867  // セッション変数アクセスの排他を行なうと、同一端末からのアクセスの処理はシリアライズされる。
868  if(page.UseSession){
869  lock(sess) {
870  page.SetSession(sess, m_appl);
871  page.PageLoad(path.Substring(kv.Key.Length));
872  }
873  } else {
874  page.SetSession(sess, m_appl);
875  page.PageLoad(path.Substring(kv.Key.Length));
876  }
877  status = 200;
878  }
879  } else {
880  status = 405;
881  }
882  break;
883  } finally {
884  if(timeout > 0)
885  NThread.SuspendWatchdog();
886  lock(m_activepages_mutex) {
887  if(m_activepages > 0)
888  m_activepages--;
889  }
890  }
891  }
892  }
893  }
894 
895  if(status == 0){
896  // ルート静的ページを探す
897  if(m_staticpagelist.ContainsKey("/")){
898  HttpStaticPage pg = m_staticpagelist["/"];
899  if(context.Request.HttpMethod == "GET"){
900  lock(pg){
901  pg.SetLogger(this.Logger);
902  pg.SetServerContext(this, context, path);
903  pg.PageLoad(path);
904  }
905  status = 200;
906  }else{
907  status = 405;
908  }
909  }
910  }
911 
912  if(status == 0)
913  // 該当ページが無い。
914  status = 404;
915 
916  } catch(PageRedirectException e) {
917  using(HttpPage page = new HttpPage()){
918  page.SetLogger(this.Logger);
919  page.SetServerContext(this, context, path);
920  page.Redirect(e.m_pagename);
921  status = 200;
922  }
923  } catch(HttpListenerException) {
924  // コンテンツ送信失敗。きっとクライアント側が切ったのだろう。
925  status = 200; // とりあえず、こちら側の処理は成功したことにする。
926  LOG_ERR("Client closed socket while sending contents.");
927  } catch(IOException e) {
928  // コンテンツ送信中にソケットが切れた(？)
929  status = 200; // とりあえず、こちら側の処理は成功したことにする。
930  LOG_DEBUG("IOError while sending contents: "+e.Message);
931  } catch(ThreadInterruptedException) {
932  using(HttpPage page = new HttpPage()){
933  page.SetLogger(this.Logger);
934  page.SetServerContext(this, context, "error");
935  page.PageLoad("503");
936  }
937  } catch(ThreadAbortException ex) {
938  bool isTimeout = (ex.ExceptionState != null)&&(ex.ExceptionState is TimeoutException);
939  if(isTimeout) {
940  LOG_ERR("{0} is aborted due to timeout", path);
941  } else {
942  LOG_ERR("{0} is aborted", path);
943  }
944  LOG_EXCEPTION(ex);
945  using(HttpPage page = new HttpPage()){
946  page.SetLogger(this.Logger);
947  page.SetServerContext(this, context, "error");
948  page.PageLoad(isTimeout?"408":"500");
949  }
950  } catch(Exception e) {
951  LOG_CRIT(String.Format("{0}: {1} in {2}", e.GetType().Name, e.Message, e.TargetSite));
952  LOG_EXCEPTION(e);
953  if(ShowStackTrace) {
954  using(HttpStackTracePage page = new HttpStackTracePage(e)) {
955  try {
956  page.SetLogger(this.Logger);
957  page.SetServerContext(this, context, "StackTrace");
958  page.PageLoad(e.Message);
959  } catch(Exception) {
960  // I can't do any more.
961  }
962  }
963  status = 200;
964  } else {
965  status = 500;
966  }
967  }
968  if(status != 200){
969  LOG_WARNING(string.Format("Failed to dispatch {0} (code={1})", requestmemo, status));
970  using(HttpPage page = new HttpPage()){
971  try {
972  page.SetLogger(this.Logger);
973  page.SetServerContext(this, context, "error");
974  page.PageLoad(status.ToString());
975  } catch(Exception) {
976  // contextがもう使えない状態にあるらしい。きっとクライアント側が切ったのだろう。
977  // 無視。
978  }
979  }
980  }
981  //context.Response.OutputStream.Close(); // 念のため。
982  // 上の行は、なぜかDebian版monoでは戻ってこない。
983  try {
984  context.Response.Close();
985  } catch(Exception) {
986  // Close失敗は無視する。
987  }
988 
989  wc.Timer.Stop();
990  long elapsed = wc.Timer.ElapsedMilliseconds;
991  if(elapsed > wc.MaxElapsed)
992  wc.MaxElapsed = elapsed;
993  wc.TotalElapsed += elapsed;
994  LOG_DEBUG(string.Format("Finished {0} ({1}msec)",requestmemo,elapsed));
995  }
