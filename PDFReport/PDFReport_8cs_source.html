450  /// <summary>
451  /// 複数のPDFファイルを結合して1つのPDFファイルを作成する
452  /// </summary>
453  /// <param name="outfilename">作成するPDFファイル名</param>
454  /// <param name="infilenames">元のPDFファイル名（任意個数指定可）</param>
455  /// <remarks>
456  /// <para>
457  /// 元のPDFファイルを1つも指定しない場合はエラー(ArgumentException)
458  /// </para>
459  /// </remarks>
460  public static void Combine(string outfilename, params string[] infilenames) {
461  if(String.IsNullOrEmpty(outfilename))
462  throw new ArgumentException("Output PDF file name is null or empty.");
463  if((infilenames == null) || (infilenames.Length <= 0))
464  throw new ArgumentException("Input PDF file is not specified.");
465  try {
466  using(FileStream fs = FileUtil.BinaryWriter(outfilename)) {
467  PdfCopyFields copy = new PdfCopyFields(fs);
468  foreach(string infilename in infilenames) {
469  if(String.IsNullOrEmpty(infilename))
470  throw new ArgumentException("Input PDF file name is null or empty.");
471  if(!File.Exists(infilename))
472  throw new IOException(String.Format("Input PDF file ({0}) does not exist.", infilename));
473  PdfReader reader = new PdfReader(infilename);
474  copy.AddDocument(reader);
475  }
476  copy.Close();
477  }
478  } catch(Exception e) {
479  File.Delete(outfilename);
480  throw e;
481  }
482  }

500  /// <summary>
501  /// フォントファイルを指定しベースフォントを得る
502  /// </summary>
503  private static BaseFont GetBaseFont(string fontfile) {
504  if(String.IsNullOrEmpty(fontfile))
505  fontfile = DefaultFontFile;
506  try {
507  string ff = fontfile.ToLower();
508  foreach(string f in basicFonts) {
509  if(ff == f.ToLower())
510  return BaseFont.CreateFont(f, BaseFont.CP1252, BaseFont.NOT_EMBEDDED);
511  }
512  return BaseFont.CreateFont(Path.Combine(FontDir, fontfile),
513  BaseFont.IDENTITY_H,
514  BaseFont.EMBEDDED);
515  } catch(Exception e) {
516  Console.WriteLine(e.Message);
517  return BaseFont.CreateFont(Path.Combine(FontDir, DefaultFontFile),
518  BaseFont.IDENTITY_H,
519  BaseFont.EMBEDDED);
520  }

540  private void calcPages(XmlFile xml) {
541  // ドキュメント内のリストフィールドの行数を数える。
542  // 注: tableタグにrows属性が付いていないものは数えられない。
543  listRows = new List<int>();
544  foreach(List<DataArray> list in listList)
545  listRows.Add(0);
546  foreach(XmlNode node in xml.Document.GetElementsByTagName("table")){
547  XmlElement elem = (XmlElement)node;
548  int listNumber = StringUtil.ToInt(elem.GetAttribute("list"), 1);
549  int rows = StringUtil.ToInt(elem.GetAttribute("rows"))+StringUtil.ToInt(elem.GetAttribute("rowoffset"), 1)-1;
550  if((listNumber <= 0) || (listNumber > listRows.Count))
551  continue;
552  if(listRows[listNumber-1] < rows)
553  listRows[listNumber-1] = rows;
554  }
555  // 最大ページ数を探す
556  totalPages = 1;
557  for(int i = 0; i < listRows.Count; i++) {
558  if(listList[i] == null)
559  continue;
560  if(listRows[i] <= 0)
561  continue;
562  int n = (listList[i].Count+listRows[i]-1)/listRows[i];
563  if(n > totalPages)
564  totalPages = n;
565  }
566  }

568  private void begin(XmlFile xml, Stream fs) {
569  if(doc != null)
570  throw new InvalidOperationException("PDF document is already open.");
571 
572  // デフォルトフォントのセットアップ
573  baseFont = GetBaseFont(DefaultFontFile);
574  fontDict = new Dictionary<string,FontParam>();
575  fontDict.Add("default", new FontParam(baseFont, 12.0f));
576  //BaseFont.AddToResourceSearch("iTextAsian.dll");
577  //Font font = new Font(BaseFont.CreateFont("HeiseiMin-W3", "UniJIS-UCS2-HW-H",false),20);
578 
579  // ドキュメント属性
580  string paper = xml.Root.GetAttribute("paper");
581  if(String.IsNullOrEmpty(paper))
582  paper = "A4";
583  bool isLandscape = ("landscape" == xml.Root.GetAttribute("orientation"));
584  Rectangle paperrect = getPageSize(paper);
585  float margin = getSize(xml.Root.GetAttribute("margin"), 36.0f);
586  float margint = getSize(xml.Root.GetAttribute("margin-top"), margin);
587  float marginr = getSize(xml.Root.GetAttribute("margin-right"), margin);
588  float marginb = getSize(xml.Root.GetAttribute("margin-bottom"), margin);
589  float marginl = getSize(xml.Root.GetAttribute("margin-left"), margin);
590  if(Rotate) {
591  isLandscape = !isLandscape;
592  float tmp = margint;
593  margint = marginr;
594  marginr = marginb;
595  marginb = marginl;
596  marginl = tmp;
597  }
598  if(isLandscape)
599  paperrect = paperrect.Rotate();
600 
601  // ドキュメントオブジェクト作成
602  doc = new Document(paperrect, marginl, marginr, margint, marginb);
603  if(DebugPaperSize) {
604  Console.WriteLine("PAPER: [{0},{1}]-[{2},{3}]", paperrect.Left, paperrect.Top, paperrect.Right, paperrect.Bottom);
605  Console.WriteLine("MARGIN: Left={0}, Top={1}, Right={2}, Bottom={3}",
606  doc.LeftMargin, doc.TopMargin, doc.RightMargin, doc.BottomMargin);
607  Console.WriteLine("REGION: [{0},{1}]-[{2},{3}]",
608  doc.Left, doc.Top, doc.Right, doc.Bottom);
609  }
610  writer = PdfWriter.GetInstance(doc, fs);
611  doc.Open();
612  }

616  XmlElement elem;
617 
618  if(doc == null)
619  throw new InvalidOperationException("PDF document has not been opened.");
620 
621  // フォント定義
622  if((elem = xml.GetElement(xml.Root, "basefont")) != null) {

733  public string Replace(Match m) {
734  string f = m.Groups[1].Value;
735  string fieldname = f;
736  string fieldopt = "";
737  int idx = f.IndexOf(':');
738  if(idx >= 0) {
739  fieldname = f.Substring(0, idx);
740  fieldopt = f.Substring(idx+1);
741  }
742  if((report.field != null) && report.field.Contains(fieldname))
743  return applyOption(report.field[fieldname], fieldopt);
744  if(fieldname.StartsWith("field")) {
745  int n = StringUtil.ToInt(fieldname.Substring(5));
746  if((report.field == null) || (n <= 0) || (n > report.field.Length))
747  return "";
748  return applyOption(report.field[n-1], fieldopt);
749  }
750  if(fieldname.StartsWith("list")) {
751  string[] subnumbers = fieldname.Substring(4).Trim("-.".ToCharArray()).Split("-.".ToCharArray());
752  int listnum, linenum;
753  string columnname;
754  switch(subnumbers.Length) {
755  case 3:
756  // リスト要素直接指定
757  listnum = StringUtil.ToInt(subnumbers[0]);
758  linenum = StringUtil.ToInt(subnumbers[1]);
759  columnname = subnumbers[2];
760  break;
761  case 2:
762  // リスト番号とカラム指定
763  listnum = StringUtil.ToInt(subnumbers[0]);
764  linenum = currentLine;
765  columnname = subnumbers[1];
766  break;
767  case 1:
768  // カラムのみ指定
769  listnum = currentList;
770  linenum = currentLine;
771  columnname = subnumbers[0];
772  break;
773  default:
774  return ""; // invalid format
775  }
776  if((report.listList == null) || (listnum <= 0) || (listnum > report.listList.Count))
777  return ""; // no such list
778  List<DataArray> list = report.listList[listnum-1];
779  int n = linenum+(report.pageNumber-1)*report.listRows[listnum-1];
780  if((list == null) || (n <= 0) || (n > list.Count))
781  return ""; // no such line
782  DataArray values = list[n-1];
783  if(values == null)
784  return ""; // empty data
785  if(values.Contains(columnname))
786  return applyOption(values[columnname], fieldopt);
787  int columnnum = StringUtil.ToInt(columnname);
788  if((columnnum <= 0) || (columnnum > values.Length))
789  return ""; // no such column
790  return applyOption(values[columnnum-1], fieldopt);
791  }
792  switch(fieldname){
793  case "page":
794  return applyOption(report.pageNumber.ToString(), fieldopt);
795  case "totalpages":
796  return applyOption(report.totalPages.ToString(), fieldopt);
797  case "currentdate":
798  return applyOption(DateTime.Now.ToString("yyyy/MM/dd"), fieldopt);
799  case "currenttime":
800  return applyOption(DateTime.Now.ToString("HH:mm:ss"), fieldopt);
801  case "firstpage":
802  return (report.pageNumber == 1)?"yes":"no";
803  case "lastpage":
804  return (report.pageNumber == report.totalPages)?"yes":"no";
805  case "lastline":
806  if((report.listList == null) || (currentList <= 0) || (currentList > report.listList.Count)) {
807  return "no";
808  } else {
809  List<DataArray> list = report.listList[currentList-1];
810  if(list == null)
811  return "no";
812  return (currentLine == list.Count)?"yes":"no";
813  }
814  }
815  return m.Value;
816  }

823  private static string applyOption(string txt, string opt) {
824  if(txt == null)
825  return null;
826  if(txt == "")
827  return "";
828  foreach(string o in opt.Split(":".ToCharArray())) {
829  if(o == "")
830  continue;
831  string oname = o;
832  string[] arg = null;
833  int idx = o.IndexOf('(');
834  if((idx >= 0) && (o[o.Length-1] == ')')) {
835  oname = o.Substring(0, idx);
836  arg = o.Substring(idx+1,o.Length-idx-2).Split(",".ToCharArray());
837  for(int i = 0; i < arg.Length; i++)
838  arg[i] = arg[i].Trim();
839  }
840  oname = oname.Trim().ToLower();
841  switch(oname) {
842  case "number": // 数値表記指定
843  if((arg != null) && (arg.Length >= 1)) {
844  // 小数点下桁数指定
845  txt = StringUtil.ToDouble(txt).ToString("F"+StringUtil.ToInt(arg[0]).ToString());
846  } else {
847  // 整数指定
848  txt = StringUtil.ToDouble(txt).ToString("F0");
849  }
850  break;
851  case "decimal": // 位取り表記指定
852  if((arg != null) && (arg.Length >= 1)) {
853  // 小数点下桁数指定
854  txt = StringUtil.DecimalString(txt, StringUtil.ToInt(arg[0]));
855  } else {
856  // 整数指定
857  txt = StringUtil.DecimalString(txt, 0);
858  }
859  break;
860  case "fixlength": // 固定長表記
861  {
862  int len = 5;
863  char fill = '0';
864  if(arg != null) {
865  if(arg.Length >= 1)
866  len = StringUtil.ToInt(arg[0],len);
867  if((arg.Length >= 2) && (arg[1].Length >= 1))
868  fill = arg[1][0];
869  }
870  txt = StringUtil.FixLength(txt, len, fill);
871  }
872  break;
873  case "plusonly": // 正の数値のみ
874  if(StringUtil.ToDouble(txt, -1.0) < 0.0)
875  txt = "0";
876  break;
877  case "prefix": // 前置文字列
878  if((arg != null) && (arg.Length >= 1))
879  txt = arg[0]+txt;
880  break;
881  case "postfix": // 前置文字列
882  if((arg != null) && (arg.Length >= 1))
883  txt = txt+arg[0];
884  break;
885  }
886  }
887  return txt;
888  }
889  }
