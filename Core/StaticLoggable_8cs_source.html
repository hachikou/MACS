130  private static string className() {
131  StackTrace st = new StackTrace(false);
132  if(st.FrameCount < 3)
133  return "";
134  StackFrame sf = st.GetFrame(2);
135  MethodBase method = sf.GetMethod();
136  Type type = method.DeclaringType;
137  return type.Name;
138  }
