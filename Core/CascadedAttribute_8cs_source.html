39  /// <summary>
40  /// 属性値取得
41  /// </summary>
42  public string Get(string name, string defValue=null, int depth=0) {
43  if(depth >= 256) // 循環参照防止
44  return defValue;
45  string val;
46  // XML属性に定義があればそれを利用する
47  if(dict.TryGetValue(name, out val)) {
48  return val;
49  }
50  if(parent != null) {
51  // クラス指定があれば親属性からクラス属性を獲得する
52  if(dict.TryGetValue("class", out val)) {
53  val = parent.getClassAttribute(val, name, depth+1);
54  if(val != null)
55  return val;
56  }
57  // デフォルトクラス属性を獲得する（トップレベルで呼ばれた時のみ）
58  if(depth == 0) {
59  if(!String.IsNullOrEmpty(defaultClassName)) {
60  val = parent.getClassAttribute(defaultClassName, name, depth+1);
61  if(val != null) {
62  return val;
63  }
64  }
65  val = parent.getClassAttribute("default", name, depth+1);
66  if(val != null) {
67  return val;
68  }
69  }
70  }
71  return defValue;
72  }

191  private void loadClass(XmlElement elem) {
192  // クラス定義を取り込む
193  foreach(XmlNode node in elem.ChildNodes) {
194  if((node.NodeType == XmlNodeType.Element) && ((node.Name == "class") || (node.Name == "default"))) {
195  CascadedAttribute classAttr = CreateChild((XmlElement)node);
196  string className = (node.Name == "default")?"default":classAttr.Name;
197  if(!String.IsNullOrEmpty(className)) {
198  classDict[className] = classAttr;
199  }
200  }
201  }
202  }

204  /// <summary>
205  /// クラス属性値取得
206  /// </summary>
207  private string getClassAttribute(string className, string name, int depth=0) {
208  if(depth >= 256) // 循環参照防止
209  return null;
210  CascadedAttribute classAttr;
211  if(!String.IsNullOrEmpty(className)) {
212  // 列挙されているクラス名を順に探す
213  foreach(string cn in StringUtil.SplitCommand(className)) {
214  if(classDict.TryGetValue(cn, out classAttr)) {
215  string val = classAttr.Get(name, null, depth+1);
216  if(val != null)
217  return val;
218  }
219  if(parent != null) {
220  // 親のクラス定義を探す
221  string val = parent.getClassAttribute(cn, name, depth+1);
222  if(val != null)
223  return val;
224  }
225  }
226  }
227  // この属性のクラス定義を探す
228  if(parent != null){
229  string childClassName = this.Get(childClassKey+"class", null, depth+1);
230  if(!String.IsNullOrEmpty(childClassName))
231  return parent.getClassAttribute(childClassName, name, depth+1);
232  }
233  return null;
234  }

236  private CascadedAttribute getClass(string className) {
237  CascadedAttribute classAttr;
238  if(classDict.TryGetValue(className, out classAttr)) {
239  return classAttr;
240  }
241  if(parent != null) {
242  return parent.getClass(className);
243  }
244  return null;
245  }

247  private bool hasClass(string className, int depth) {
248  if(depth >= 256)
249  return false;
250  string val;
251  if((depth == 0) && (defaultClassName == className))
252  return true;
253  if(!dict.TryGetValue("class", out val))
254  return false;
255  foreach(string cname in StringUtil.SplitCommand(val)) {
256  if(cname == className)
257  return true;
258  CascadedAttribute xclass = getClass(cname);
259  if((xclass != null) && xclass.hasClass(className, depth+1))
260  return true;
261  }
262  return false;
263  }
