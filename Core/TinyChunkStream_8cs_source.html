132  public override int Read(byte[] buf, int offset, int size) {
133  if(writing)
134  throw new InvalidOperationException("Can't read TinyChunkStream after write.");
135  if(chunkSize == 0)
136  return 0;
137  if(buffer == null) {
138  buffer = new byte[255];
139  count = 0;
140  }
141  int len = 0;
142  while(size > 0) {
143  if(count == 0) {
144  if(baseStream.Read(buffer,0,1) != 1)
145  break;
146  chunkSize = count = buffer[0];
147  if(chunkSize == 0)
148  break;
149  }
150  int sz = (int)count;
151  if(size < sz)
152  sz = size;
153  Buffer.BlockCopy(buffer,chunkSize-count,buf,offset,sz);
154  size -= sz;
155  offset += sz;
156  count -= (byte)sz;
157  len += sz;
158  }
159  return len;
160  }

168  public override void Write(byte[] buf, int offset, int size) {
169  if(buffer == null) {
170  buffer = new byte[chunkSize];
171  count = 0;
172  }
173  writing = true;
174  while(size > 0) {
175  int sz = (int)(chunkSize-count);
176  if(size < sz)
177  sz = size;
178  Buffer.BlockCopy(buf,offset,buffer,count,sz);
179  size -= sz;
180  offset += sz;
181  count += (byte)sz;
182  if(count >= chunkSize){
183  Flush(false);
184  }
185  }
186  }

192  public void Flush(bool baseFlush) {
193  if(!writing || (count == 0) || (buffer == null))
194  return;
195  byte[] b = new byte[1];
196  b[0] = count;
197  baseStream.Write(b,0,1);
198  baseStream.Write(buffer,0,count);
199  if(baseFlush)
200  baseStream.Flush();
201  count = 0;
202  }
203 
204  public void Finish(bool baseFlush=false) {
205  if(!writing || (buffer == null))
206  return;
207  Flush(false);
208  byte[] b = new byte[1];
209  b[0] = 0;
210  baseStream.Write(b,0,1);
211  if(baseFlush)
212  baseStream.Flush();
213  buffer = null;
214  }
