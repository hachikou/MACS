44  /// <summary>
45  /// HTMLエンコーディングした文字列を返す。
46  /// </summary>
47  public static string HE(string txt) {
48  if(txt == null)
49  return "";
50  //return HttpUtility.HtmlEncode(txt); // HtmlEncodeは日本語も&#変換してしまう。
51  StringBuilder sb = new StringBuilder();
52  foreach(char ch in txt){
53  switch(ch){
54  case '&':
55  sb.Append("&amp;");
56  break;
57  case '<':
58  sb.Append("&lt;");
59  break;
60  case '>':
61  sb.Append("&gt;");
62  break;
63  case '\"':
64  sb.Append("&quot;");
65  break;
66  case '\'':
67  sb.Append("&#39;");
68  break;
69  default:
70  sb.Append(ch);
71  break;
72  }
73  }
74  return sb.ToString();
75  }

86  /// <summary>
87  /// クォート文字をエスケープした文字列を返す。
88  /// </summary>
89  public static string QE(string txt) {
90  if(txt == null)
91  return "";
92  StringBuilder sb = new StringBuilder();
93  foreach(char ch in txt) {
94  bool flag = false;
95  foreach(KeyValuePair<char,string> kv in QEDict) {
96  if(ch == kv.Key) {
97  sb.Append(kv.Value);
98  flag = true;
99  break;
100  }
101  }
102  if(!flag)
103  sb.Append(ch);
104  }
105  return sb.ToString();
106  //return txt.Replace("&", "&amp;").Replace("\\", "&#92;").Replace("\"", "&quot;").Replace("'", "&#39;");
107  }

109  private static readonly Dictionary<char,string> QEDict = new Dictionary<char,string>() {
110  {'\n', "\\n"},
111  {'\f', "\\f"},
112  {'\b', "\\b"},
113  {'\r', "\\r"},
114  {'\t', "\\t"},
115  {'\'', "\\'"},
116  {'"', "\\\""},
117  {'\\',"\\\\"}
118  };
