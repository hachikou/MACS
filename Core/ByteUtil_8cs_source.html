510  /// <summary>
511  /// バイト列を書き込む。
512  /// コピー元バイト列の長さが必要サイズ未満の場合には、fillで埋められる。
513  /// </summary>
514  public static void ToBytes(byte[] val, byte[] buf, int offset, int len, byte fill=0) {
515  if(val == null) {
516  if(fill == 0)
517  Array.Clear(buf, offset, len);
518  else
519  for(int i = 0; i < len; i++)
520  buf[offset+i] = fill;
521  return;
522  }
523  if(val.Length < len){
524  Buffer.BlockCopy(val, 0, buf, offset, val.Length);
525  if(fill == 0)
526  Array.Clear(buf, offset+val.Length, len-val.Length);
527  else
528  for(int i = 0; i < len-val.Length; i++)
529  buf[offset+val.Length+i] = fill;
530  } else {
531  Buffer.BlockCopy(val, 0, buf, offset, len);
532  }
533  }
