277  /// <summary>
278  /// 指定した行の間を文字列化して返す。
279  /// </summary>
280  /// <remarks>
281  /// <para>
282  /// 行指定は、先頭行が0である。
283  /// </para>
284  /// </remarks>
285  public string ToString(int startline, int endline) {
286  StringBuilder xsb = new StringBuilder();
287  if(startline < 0)
288  startline = 0;
289  if((endline <= 0) || (endline > buffer.Count))
290  endline = buffer.Count;
291  int lineno = startLineNumber+startline;
292  int xlines = 0;
293  for(int i = startline; i < endline; i++) {
294  string line = buffer[i];
295  if((maxLines > 0) && (xlines >= maxLines))
296  break;
297  if(lineNumberFormat == null)
298  xsb.Append(lineno);
299  else
300  xsb.Append(String.Format(lineNumberFormat, lineno));
301  xsb.Append(line);
302  lineno++;
303  xlines++;
304  }
305  return xsb.ToString();
306  }
