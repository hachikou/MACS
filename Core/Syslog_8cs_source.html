406  private void setServer(string name) {
407  if(String.IsNullOrEmpty(name)) {
408  server = "localhost";
409  return;
410  }
411  string[] x = name.Split(":".ToCharArray(),2);
412  if(x.Length > 1) {
413  server = x[0];
414  port = StringUtil.ToInt(x[1]);
415  } else {
416  server = name;
417  }
418  }

436  private void log(Level level, string msg, params object[] args) {
437  if(level > SendLevel)
438  return;
439  // RFC3164に従ったメッセージを作る
440  if(HostName == null)
441  HostName = Dns.GetHostName().Split(".".ToCharArray())[0];
442  DateTime dt = DateTime.Now;
443  string formatted;
444  if(String.IsNullOrEmpty(ModuleName))
445  formatted = "";
446  else
447  formatted = ModuleName+" ";
448  try {
449  formatted += String.Format(msg, args);
450  } catch(FormatException) {
451  formatted += msg;
452  }
453  string txt = String.Format("<{0}>{1} {2,2} {3:D2}:{4:D2}:{5:D2} {6} {7}",
454  (int)Facility+(int)level,
455  monthName[dt.Month-1], dt.Day, dt.Hour, dt.Minute, dt.Second,
456  HostName,
457  formatted);
458  byte[] buf = enc.GetBytes(txt);
459  if(udp == null)
460  open();
461  if(udp == null)
462  return;
463  udp.Send(buf, buf.Length);
464  }
