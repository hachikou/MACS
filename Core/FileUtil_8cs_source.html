582  /// <summary>
583  /// ファイル名ワイルドカード文字(*,?)を展開してマッチするファイル名の一覧を返す
584  /// </summary>
585  public static string[] ExpandWildCard(string path, bool ignoreCase=true) {
586  //Console.WriteLine("ExpandWildCard: '{0}'", path);
587  List<string> res = new List<string>();
588  string dir = Path.GetDirectoryName(path);
589  string file = Path.GetFileName(path);
590  //Console.WriteLine("ExpandWildCard: dir='{0}', file='{1}'", dir, file);
591  if(!String.IsNullOrEmpty(dir) && (dir.Contains("*") || dir.Contains("?"))) {
592  foreach(string d in ExpandWildCard(dir)) {
593  if (Directory.Exists(d)) {
594  foreach (string x in ExpandWildCard(Path.Combine(d, file))) {
595  res.Add(x);
596  }
597  }
598  }
599  return res.ToArray();
600  }
601  if(String.IsNullOrEmpty(dir))
602  dir = ".";
603  if(!Directory.Exists(dir))
604  return res.ToArray();
605  string pat_str = "^"+Regex.Escape(file).Replace(@"\*",".*").Replace(@"\?",".")+"$";
606  Regex pat;
607  if(ignoreCase)
608  pat = new Regex(pat_str, RegexOptions.CultureInvariant|RegexOptions.IgnoreCase);
609  else
610  pat = new Regex(pat_str);
611  try {
612  foreach(string d in Directory.EnumerateDirectories(dir))
613  if(pat.IsMatch(Path.GetFileName(d)))
614  res.Add(d);
615  foreach(string f in Directory.EnumerateFiles(dir))
616  if(pat.IsMatch(Path.GetFileName(f)))
617  res.Add(f);
618  } catch(UnauthorizedAccessException) {
619  // just ignore.
620  }
621  return res.ToArray();
622  }

641  /// </summary>

643  /// <param name="depth">サブディレクトリをたどる深さ。0の場合サブディレクトリをたどらない</param>

646  /// <para>

648  /// </para>
649  /// </remarks>
650  public static long GetDirectorySize(DirectoryInfo di, int depth=99) {
651  long totalSize = 0;
652  try {
653  foreach(FileInfo fi in di.EnumerateFiles()) {
654  totalSize += fi.Length;
655  }
656  if(depth <= 0)
657  return totalSize;
658  foreach(DirectoryInfo fi in di.EnumerateDirectories()) {
659  totalSize += GetDirectorySize(fi, depth-1);
660  }
661  } catch(Exception) {
662  // just ignore.
663  }
664  return totalSize;
665  }
