15 /// <summary>
16 /// ReaderWriterLockSlimをデバッグしやすくするためのラッピングクラス
17 /// </summary>
18 /// <remarks>
19 /// <para>
20 /// lock(mutex){...} と同じような感覚でRead/Writeロックを使えるようにしたクラスです。
21 /// RWLockオブジェクトは、上記の'mutex'にあたる、ロック対象オブジェクトで、
22 /// その実態は C#標準の ReaderWriterLockSlimです。
23 /// 従来lock()を使って記述していたコードを、次のように記述することでRead/Write
24 /// ロック化することができます。
25 ///
26 /// object mutex = new object();
27 /// .....
28 /// lock(mutex) {
29 /// .....
30 /// }
31 ///
32 /// ↓↓↓
33 /// RWLock rwlock = new RWLock("ロック名称");
34 /// .....
35 /// // Read lockをする場合
36 /// using(var lockhandle = new ReadLock(rwlock)) {
37 /// .....
38 /// }
39 /// // Write lockをする場合
40 /// using(var lockhandle = new WriteLock(rwlock)) {
41 /// .....
42 /// }
43 ///
44 /// "ロック名称"は、デバッグを容易にするための任意の文字列で、ロック動作には
45 /// 関係ありません。
46 /// ReadLockまたはWriteLockインスタンスをnewするところでロックがかかり（必要
47 /// に応じて待ちが発生し）、using句を抜けてインスタンスがDisposeされる際に
48 /// ロックが解除されます。
49 /// using句を使って確実にロックを解除できるので、ReaderWriterLockSlimクラス
50 /// をそのまま使う時に起こりがちな、ロック解除コードの書き忘れを防ぐことが
51 /// できます。
52 ///
53 /// RWLock.DumpAll(OpeLog logger) メソッドを使うと、現在使われている全ての
54 /// RWLockの状態をログに書き出すことができます。
55 /// </para>
56 /// </remarks>
57 public class RWLock : ReaderWriterLockSlim {

132  private void dump(OpeLog logger, TextWriter w, bool full) {
133  if(!full && (this.CurrentReadCount == 0) && !this.IsWriteLockHeld && !this.IsUpgradeableReadLockHeld)
134  return;
135  if(logger != null) {
136  logger.Log("RWLock["+this.Name+"]", OpeLog.Level.DEBUG,
137  "reading {0}, writing {1}, u-reading {2}",
138  this.CurrentReadCount, this.IsWriteLockHeld?1:0, this.IsUpgradeableReadLockHeld?1:0);
139  }
140  if(w != null) {
141  w.WriteLine("RWLock[{0}]: reading {1}, writing {2}, u-reading {3}",
142  this.Name, this.CurrentReadCount, this.IsWriteLockHeld?1:0, this.IsUpgradeableReadLockHeld?1:0);
143  }
144  if((readingThread != null) && (readingThread.Count > 0)) {
145  StringBuilder sb = new StringBuilder("reading:");
146  foreach(Thread t in readingThread) {
147  sb.Append(' ');
148  sb.Append(String.IsNullOrEmpty(t.Name)?t.ManagedThreadId.ToString():t.Name);
149  }
150  if(logger != null)
151  logger.Log("RWLock["+this.Name+"]", OpeLog.Level.DEBUG, sb.ToString());
152  if(w != null)
153  w.WriteLine("RWLock[{0}]: {1}", this.Name, sb.ToString());
154  }
155  if((writingThread != null) && (writingThread.Count > 0)) {
156  StringBuilder sb = new StringBuilder("writing:");
157  foreach(Thread t in writingThread) {
158  sb.Append(' ');
159  sb.Append(String.IsNullOrEmpty(t.Name)?t.ManagedThreadId.ToString():t.Name);
160  }
161  if(logger != null)
162  logger.Log("RWLock["+this.Name+"]", OpeLog.Level.DEBUG, sb.ToString());
163  if(w != null)
164  w.WriteLine("RWLock[{0}]: {1}", this.Name, sb.ToString());
165  }
166  if((upgradeableReadingThread != null) && (upgradeableReadingThread.Count > 0)) {
167  StringBuilder sb = new StringBuilder("upgradeableReading:");
168  foreach(Thread t in upgradeableReadingThread) {
169  sb.Append(' ');
170  sb.Append(String.IsNullOrEmpty(t.Name)?t.ManagedThreadId.ToString():t.Name);
171  }
172  if(logger != null)
173  logger.Log("RWLock["+this.Name+"]", OpeLog.Level.DEBUG, sb.ToString());
174  if(w != null)
175  w.WriteLine("RWLock[{0}]: {1}", this.Name, sb.ToString());
176  }
177  if((waitingReadThread != null) && (waitingReadThread.Count > 0)) {
178  StringBuilder sb = new StringBuilder("waiting read:");
179  foreach(Thread t in waitingReadThread) {
180  sb.Append(' ');
181  sb.Append(String.IsNullOrEmpty(t.Name)?t.ManagedThreadId.ToString():t.Name);
182  }
183  if(logger != null)
184  logger.Log("RWLock["+this.Name+"]", OpeLog.Level.DEBUG, sb.ToString());
185  if(w != null)
186  w.WriteLine("RWLock[{0}]: {1}", this.Name, sb.ToString());
187  }
188  if((waitingWriteThread != null) && (waitingWriteThread.Count > 0)) {
189  StringBuilder sb = new StringBuilder("waiting read:");
190  foreach(Thread t in waitingWriteThread) {
191  sb.Append(' ');
192  sb.Append(String.IsNullOrEmpty(t.Name)?t.ManagedThreadId.ToString():t.Name);
193  }
194  if(logger != null)
195  logger.Log("RWLock["+this.Name+"]", OpeLog.Level.DEBUG, sb.ToString());
196  if(w != null)
197  w.WriteLine("RWLock[{0}]: {1}", this.Name, sb.ToString());
198  }
199  if((waitingUpgradeableReadThread != null) && (waitingUpgradeableReadThread.Count > 0)) {

238  private static void dumpAll(OpeLog logger, TextWriter w, bool full) {
239  RWLock[] list;
240  lock(globalRWLockList) {
241  list = globalRWLockList.ToArray();
242  }
243  Array.Sort(list, delegate(RWLock a, RWLock b){
244  return a.Name.CompareTo(b.Name);
245  });
246  if(logger != null)
247  logger.Log("RWLock", OpeLog.Level.DEBUG, "total {0} locks", list.Length);
248  if(w != null)
249  w.WriteLine("RWLock: total {0} locks", list.Length);
250  foreach(RWLock rwlock in list) {
251  rwlock.dump(logger, w, full);
252  }
253  }

259  public void doEnterReadLock() {
260  if(waitingReadThread != null) {
261  lock(waitingReadThread) {
262  waitingReadThread.Add(Thread.CurrentThread);
263  }
264  EnterReadLock();
265  lock(waitingReadThread) {
266  waitingReadThread.Remove(Thread.CurrentThread);
267  }
268  } else {
269  EnterReadLock();
270  }
271  if(readingThread != null) {
272  lock(readingThread) {
273  readingThread.Add(Thread.CurrentThread);
274  }
275  }
276  }
277 
278  public void doExitReadLock() {
279  if(readingThread != null) {

282  }
283  }

354 /// <summary>
355 /// ReaderWriterLockSlimを安全に用いるためのラッピングクラス
356 /// ReadLock, WriteLock, UpgradableLockクラスの共通基底
357 /// </summary>
358 public abstract class LockHandle : Loggable, IDisposable {
359 
360  public LockHandle(RWLock rwlock_) {
361  rwlock = rwlock_;
362  }
363 
364  ~LockHandle() {
365  Dispose();
366  }
367 
368  public abstract void Dispose();
369 
370  protected RWLock rwlock;
371 }

378  public ReadLock(RWLock rwlock_) : base(rwlock_) {
