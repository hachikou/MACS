149  /// <summary>
150  /// ログを追記する
151  /// </summary>
152  public void Log(string category, Level lv, string msg, bool ignoreNewLine=true) {
153  lock(m_mutex){
154  if(lv >= m_consolelevel)
155  Console.WriteLine("{0}: {1}", category, msg);
156  if(lv < m_level)
157  return;
158  if(m_size > 0){
159  FileInfo fi = new FileInfo(m_filename);
160  if(fi.Exists && (fi.Length/1000 >= m_size)){
161  try {
162  Rotate();
163  } catch(IOException e) {
164  Console.WriteLine("Can't rotate log file {0}: {1}", m_filename, e.Message);
165  return;
166  }
167  }
168  }
169  StringBuilder txt = new StringBuilder();
170  txt.Append(DateTime.Now.ToString("yyyy/MM/dd HH:mm:ss.fff "));
171  txt.Append(category);
172  txt.Append(' ');
173  txt.Append(lv.ToString());
174  txt.Append(' ');
175  if(ignoreNewLine)
176  txt.Append(msg.Replace('\n',' ').Replace('\r',' '));
177  else
178  txt.Append(msg);
179  //Console.WriteLine(txt);
180  StreamWriter sw = FileUtil.SharedAppendWriter(m_filename, m_enc);
181  if(sw == null)
182  return;
183  sw.WriteLine(txt.ToString());
184  sw.Close();
185  }
186  }

195  /// <summary>
196  /// ローテートする
197  /// </summary>
198  public void Rotate() {
199  removeOld();
200  string basename = Path.Combine(Path.GetDirectoryName(m_filename), Path.GetFileNameWithoutExtension(m_filename))+".";
201  string ext = Path.GetExtension(m_filename);
202  int n = m_ages;
203  while(n > 0){
204  string fname = basename+n.ToString()+ext;
205  string xfname;
206  if(n > 1)
207  xfname = basename+(n-1).ToString()+ext;
208  else
209  xfname = m_filename;
210  File.Delete(fname);
211  if(File.Exists(xfname))
212  File.Move(xfname, fname);
213  n--;
214  }
215  File.Delete(m_filename);
216  Log("LOGGER", Level.INFO, "Operation log file is rotated.");
217  }
