167  /// <summary>
168  /// メールを送信する。
169  /// </summary>
170  public bool Send(bool exceptionFlag=false) {
171  lock(mutex) {
172  RemoteCertificateValidationCallback orgCertificationCallback = ServicePointManager.ServerCertificateValidationCallback;
173  try {
174  using(SmtpClient smtp = new SmtpClient(Server, Port)) {
175  smtp.Timeout = Timeout;
176  string authstring;
177  switch(Auth){
178  case AuthType.NONE:
179  authstring = "No authentification";
180  break;
181  case AuthType.POP:
182  authstring = "POP before SMTP:"+AuthUser;
183  if(!PopBeforeSmtp())
184  return false;
185  Thread.Sleep(PopWait);
186  break;
187  case AuthType.SMTP:
188  authstring = "SMTP-AUTH:"+AuthUser;
189  smtp.Credentials = new NetworkCredential(AuthUser, AuthPass);
190  break;
191  default:
192  authstring = "Unknown";
193  break;
194  }
195  smtp.EnableSsl = EnableSsl;
196  if(smtp.EnableSsl) {
197  authstring += ",STARTTLS";
198  if(IgnoreCert) {
199  ServicePointManager.ServerCertificateValidationCallback = ignoreCertificateValidation;
200  }
201  }
202  LOG_INFO(string.Format("Sending mail from {0} to {1} server={2}:{3} ({4}) timeout={5}msec", Sender.ToString(), Receivers.ToString(), Server, Port, authstring, Timeout));
203  using(MailMessage msg = new MailMessage()){
204  msg.From = Sender;
205  foreach(MailAddress a in Receivers)
206  msg.To.Add(a);
207  if(MailEncoding == Encoding.GetEncoding("iso-2022-jp")) {
208  // .Net4.5の実装異常に対応するため、次のように二重にBエンコーディングをかける
209  msg.Subject = EncodeIso2022jpBase64(EncodeIso2022jpBase64(Subject));
210  } else {
211  msg.Subject = Subject;
212  msg.SubjectEncoding = MailEncoding;
213  }
214  msg.Body = Body;
215  msg.BodyEncoding = MailEncoding;
216  smtp.Send(msg);
217  }
218  }
219  } catch(SmtpException e){
220  LOG_ERR(string.Format("Failed to send mail ({0})", e.Message));
221  if(exceptionFlag)
222  throw e;
223  return false;
224  } finally {
225  ServicePointManager.ServerCertificateValidationCallback = orgCertificationCallback;
226  }
227  }
228  return true;
229  }

231  /// <summary>
232  /// POPアクセスを試みる。
233  /// </summary>
234  protected bool PopBeforeSmtp() {
235  if(string.IsNullOrEmpty(PopServer)){
236  LOG_ERR("POP server is not specified.");
237  return false;
238  }
239  if((PopPort <= 0) || (PopPort >= 65535)){
240  LOG_ERR("Invalid POP port number.");
241  return false;
242  }
243  if(string.IsNullOrEmpty(AuthUser)){
244  LOG_ERR("POP user is not specified.");
245  return false;
246  }
247  if(string.IsNullOrEmpty(AuthPass)){
248  LOG_ERR("POP password is not specified.");
249  return false;
250  }
251  TcpClient client = new TcpClient();
252  client.ReceiveTimeout = Timeout;
253  client.SendTimeout = Timeout;
254  try {
255  client.Connect(PopServer, PopPort);
256  //LOG_DEBUG("POP connected");
257  using(NetworkStream ns = client.GetStream()){
258  ns.ReadTimeout = Timeout;
259  ns.WriteTimeout = Timeout;
260  string msg = Recv(ns);
261  //LOG_DEBUG("POP Got "+msg);
262  if(!msg.StartsWith("+OK")){
263  LOG_ERR(string.Format("Invalid responce from POP connection ({0})", msg));
264  return false;
265  }
266  Send(ns, "USER "+AuthUser);
267  msg = Recv(ns);
268  //LOG_DEBUG("POP Got "+msg);
269  if(!msg.StartsWith("+OK")){
270  LOG_ERR(string.Format("Invalid responce from POP USER command ({0})", msg));
271  return false;
272  }
273  Send(ns, "PASS "+AuthPass);
274  msg = Recv(ns);
275  //LOG_DEBUG("POP Got "+msg);
276  if(!msg.StartsWith("+OK")){
277  LOG_ERR(string.Format("POP Authentification failed ({0})", msg));
278  return false;
279  }
280  Send(ns, "QUIT");
281  ns.Close();
282  //LOG_DEBUG("POP Fin.");
283  }
284  } catch(Exception e){
285  LOG_ERR(string.Format("POP session is aborted. ({0})", e.Message));
286  return false;
287  } finally {
288  client.Close();
289  }
290  return true;
291  }

301  /// <summary>
302  /// メールサーバから1行受信する。
303  /// </summary>
304  protected string Recv(NetworkStream ns) {
305  byte[] buf = new byte[256];
306  for(int i = 0; i < buf.Length; i++){
307  int ch = ns.ReadByte();
308  if((ch < 0) || (ch == 0x0a)){
309  if((i > 0) && (buf[i-1] == 0x0d))
310  return Encoding.ASCII.GetString(buf, 0, i-1);
311  else
312  return Encoding.ASCII.GetString(buf, 0, i);
313  }
314  buf[i] = (byte)ch;
315  }
316  return Encoding.ASCII.GetString(buf);
317  }

319  /// <summary>
320  /// ISO2022JP Base64エンコーディング
321  /// </summary>
322  /// <remarks>
323  /// <para>
324  /// .NetフレームワークがISO2022JPのエンコーディングをきちんとできない対策
325  /// </para>
326  /// </remarks>
327  private static string EncodeIso2022jpBase64(string txt) {
328  string strBase64 = Convert.ToBase64String(Encoding.GetEncoding("iso-2022-jp").GetBytes(txt));
329  return string.Format("=?{0}?B?{1}?=", "iso-2022-jp", strBase64);
330  }
