89  /// <summary>
90  /// スタックトレースの文字列を得る
91  /// </summary>
92  public static string GetExceptionMessage(Exception e) {
93  StringBuilder sb = new StringBuilder();
94  string indent = "";
95  while(e != null) {
96  sb.Append(indent);
97  sb.AppendFormat("{0}: {1}\n", e.GetType().Name, e.Message);
98  sb.AppendFormat(indent);
99  sb.AppendFormat("StackTrace:\n");
100  foreach(string st in e.StackTrace.Split("\n".ToCharArray())) {
101  sb.Append(indent);
102  sb.Append(st.Replace('\r',' '));
103  sb.Append('\n');
104  }
105  e = e.InnerException;
106  if(e != null) {
107  sb.Append(indent);
108  sb.Append("InnerException:\n");
109  indent += " ";
110  }
111  }
112  return sb.ToString();
113  }

209  private string logMessage(string msg, object[] objs) {
210  StringBuilder sb = new StringBuilder();
211  string pagename = GetPageName();
212  if(pagename != null){
213  sb.Append("PAGE:");
214  sb.Append(pagename);
215  sb.Append(" ");
216  }
217  string username = GetUserName();
218  if(username != null){
219  sb.Append("USER:");
220  sb.Append(username);
221  sb.Append(" ");
222  }
223  if((objs == null) || (objs.Length == 0))
224  sb.Append(msg);
225  else
226  sb.AppendFormat(msg, objs);
227  return sb.ToString();
228  }
