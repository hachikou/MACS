287  private void write(int len, bool rightAlign, string txt, byte fillbyte, bool fillFlag, bool cutFlag) {
288  if(len <= 0)
289  return;
290  if(txt == null)
291  txt = "";
292  byte[] b = SJISDictionary.GetBytes(txt);
293  if(b.Length >= len) {
294  if(cutFlag) {
295  while(b.Length > len) {
296  txt = txt.Substring(0, txt.Length-1);
297  b = SJISDictionary.GetBytes(txt);
298  }
299  }
300  fs.Write(b, 0, b.Length);
301  int blen = b.Length;
302  b = new byte[1];
303  b[0] = 0x20;
304  for(int i = 0; i < len-blen; i++)
305  fs.Write(b, 0, 1);
306  return;
307  }
308  if(fillFlag) {
309  byte[] fill = new byte[len-b.Length];
310  for(int i = 0; i < fill.Length; i++)
311  fill[i] = 0x20;
312  if(rightAlign) {
313  fs.Write(fill, 0, fill.Length);
314  fs.Write(b, 0, b.Length);
315  } else {
316  fs.Write(b, 0, b.Length);
317  fs.Write(fill, 0, fill.Length);
318  }
319  } else {
320  fs.Write(b, 0, b.Length);
321  }
322  }
