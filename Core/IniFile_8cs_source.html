101  /// <summary>
102  /// INIファイルを書き出す。
103  /// </summary>
104  /// <param name="makebak">バックアップファイルを作るかどうか。</param>
105  /// <param name="tmpdir">一時保存ファイルを作るディレクトリ名。nullを指定したり省略した場合はINIファイルと同じディレクトリが使われる。</param>
106  /// <remarks>
107  /// <para>
108  /// 既存ファイルに書き出す場合、そのファイルに書かれている内容のうち、
109  /// 本データベースで値を持っている定義行以外のすべての行は、そのまま
110  /// 保持される。（コメント行や空行は変更されない。）
111  /// </para>
112  /// </remarks>
113  public bool Write(bool makebak=true, string tmpdir=null) {
114  lock(m_mutex){
115  if(m_filename == null)
116  return false;
117  string tmpfilename;
118  if(String.IsNullOrEmpty(tmpdir)) {
119  tmpfilename = m_filename;
120  } else {
121  tmpfilename = Path.Combine(tmpdir,Path.GetFileName(m_filename));
122  }
123  tmpfilename += "."+rnd.Next(1000000).ToString();
124  try {
125  // まず元ファイルを一時ファイルにコピーする
126  if(File.Exists(m_filename)) {
127  if(!FileUtil.Copy(m_filename, tmpfilename))
128  return false;
129  if(makebak)
130  FileUtil.Copy(m_filename, m_filename+".bak");
131  }
132  using(StreamWriter sw = FileUtil.Writer(m_filename, m_enc)){
133  if(sw == null)
134  return false;
135  if(m_data == null)
136  m_data = new Dictionary<string,string>(); // ダミーで空データを書き出す
137  if(File.Exists(tmpfilename)) {
138  using(StreamReader sr = FileUtil.Reader(tmpfilename, m_enc)){
139  if(sr == null)
140  return false;
141  while(!sr.EndOfStream) {
142  string line = sr.ReadLine().Trim();
143  if(line.StartsWith("#") || line.StartsWith(";") || line.StartsWith("[")) {
144  sw.WriteLine(line);
145  continue;
146  }
147  string[] keyval = line.Split("=".ToCharArray(), 2);
148  if(keyval.Length != 2) {
149  sw.WriteLine(line);
150  continue;
151  }
152  string key = keyval[0].Trim();
153  if(m_data.ContainsKey(key)) {
154  sw.WriteLine(key + "=" + m_data[key]);
155  m_data.Remove(key);
156  } else {
157  sw.WriteLine(line);
158  }
159  }
160  sr.Close();
161  }
162  } else {
163  if(m_sectionname != null)
164  sw.WriteLine("[" + m_sectionname + "]");
165  }
166  foreach(string key in m_data.Keys) {
167  sw.WriteLine(key + "=" + m_data[key]);
168  }
169  sw.Close();
170  }
171  } finally {
172  if(File.Exists(tmpfilename))
173  File.Delete(tmpfilename);
174  }
175  return _reload();
176  }
177  }

293  /// <summary>
294  /// 文字列からbool値を判断する
295  /// </summary>
296  /// <remarks>
297  /// <para>
298  /// valが"yes","Yes","YES","y","Y","true","True","TRUE","t","T","1"のとき、trueを、

301  /// </para>
302  /// </remarks>
303  public static bool GetBoolFromString(string val, bool def) {
304  switch(val) {
305  case "yes":
306  case "Yes":
307  case "YES":
308  case "y":
309  case "Y":
310  case "true":
311  case "True":
312  case "TRUE":
313  case "t":
314  case "T":
315  case "1":
316  return true;
317  case "no":
318  case "No":
319  case "NO":
320  case "n":
321  case "N":
322  case "false":
323  case "False":
324  case "FALSE":
325  case "f":
326  case "F":
327  case "0":
328  return false;
329  default:
330  return def;
331  }
332  }

342  private void _reload_ifneeded() {
343  if((m_data == null)
344  || ((m_filename != null) && File.Exists(m_filename) && (m_lasttime < File.GetLastWriteTime(m_filename)))){
345  _reload();
346  }

349  private bool _reload() {
350  if(m_data == null)
351  m_data = new Dictionary<string, string>();
352  else
353  m_data.Clear();
354  m_sectionname = null;
355  m_lasttime = DateTime.MinValue;
356  if((m_filename == null) || !File.Exists(m_filename))
357  return false;
358  using(StreamReader sr = FileUtil.Reader(m_filename, m_enc)){
359  if(!_reload(sr))
360  return false;
361  sr.Close();
362  }
363  m_lasttime = File.GetLastWriteTime(m_filename);
364  return true;
365  }

367  private bool _reload(StreamReader sr) {
368  if(sr == null)
369  return false;
370  while(!sr.EndOfStream) {
371  _loadline(sr.ReadLine().Trim());
372  }
373  return true;
374  }

376  private bool _loadline(string line){
377  if(line.StartsWith("#") || line.StartsWith(";"))
378  return false;
379  if(line.StartsWith("[")){
380  int idx = line.IndexOf(']');
381  if(idx > 1)
382  m_sectionname = line.Substring(1,idx-1);
383  return true;
384  }
385  string[] keyval = line.Split("=".ToCharArray(), 2);
386  if(keyval.Length != 2)
387  return false;
388  if(m_data == null)
389  m_data = new Dictionary<string, string>();
390  m_data[keyval[0].Trim()] = keyval[1].Trim();
391  return true;
392  }
