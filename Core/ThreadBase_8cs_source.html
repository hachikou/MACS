13 /// <summary>
14 /// バックグラウンド動作スレッドを持つクラス
15 /// </summary>
16 /// <remarks>
17 /// <para>
18 /// 派生クラスではRun()メソッドを定義すること。
19 /// Runメソッド内では、StopRequestを定期的にチェックし、trueならばリソースを
20 /// 開放するなどの処置をしてメソッドからreturnすること。
21 /// </para>
22 /// </remarks>
23 public abstract class ThreadBase : Loggable,IDisposable {
24 
25  /// <summary>
26  /// コンストラクタ
27  /// </summary>
28  public ThreadBase(string name_) {
29  name = name_;
30  threadMutex = new object();
31  mythread = null;
32  }
33 
34  /// <summary>
35  /// デストラクタ
36  /// </summary>
37  ~ThreadBase() {
38  Dispose();
39  }
40 
41  /// <summary>
42  /// リソース解放
43  /// </summary>
44  public virtual void Dispose() {
45  lock(threadMutex) {
46  _waitForStop(1000);
47  }
48  }
49 
50  /// <summary>
51  /// サービス名（デバッグメッセージ用）
52  /// </summary>
53  public virtual string ThreadName {
54  get { return (mythread!=null)?mythread.Name:name; }
55  }
56 
57  /// <summary>
58  /// スレッド動作を開始する
59  /// </summary>
60  /// <remarks>
61  /// <para>
62  /// 既に開始している場合は何もしない。
63  /// </para>
64  /// </remarks>
65  public void Start() {
66  lock(threadMutex) {
67  if(mythread != null)
68  return;
69  StopRequest = false;
70  mythread = new NThread(name, Run);
71  mythread.Start();
72  }
73  }
74 
75  /// <summary>
76  /// スレッド動作の停止を要求する
77  /// </summary>
78  /// <remarks>
79  /// <para>
80  /// サブスレッドに停止要求を出すだけで、停止は待たない。
81  /// </para>
82  /// </remarks>
83  public void Stop() {
84  lock(threadMutex) {
85  StopRequest = true;
86  }
87  }
88 
89  /// <summary>
90  /// スレッド動作の停止を待つ
91  /// </summary>
92  /// <param name="timelimit">最大待ち時間。ミリ秒</param>
93  public void WaitForStop(int timelimit) {
94  lock(threadMutex) {
95  _waitForStop(timelimit);
96  }
97  }

116  private void _waitForStop(int timelimit) {
117  if(mythread == null)
118  return;
119  StopRequest = true;
120  mythread.Join(timelimit);
