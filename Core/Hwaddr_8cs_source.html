62  /// 文字列からMACアドレスをセットする
63  /// </summary>
64  /// <remarks>
65  /// <para>
66  /// HEX値の区切り文字は':'、'-'または' 'のいずれでもよい。
67  /// 6バイト以下のデータも読み取る。
68  /// </para>
69  /// </remarks>
70  /// <returns>自分自身</returns>
71  public Hwaddr Set(string hwaddr) {
72  if((hwaddr == null) || (hwaddr == "")){
73  vals = new byte[0];
74  return this;
75  }
76  vals = null;
77  byte[] tmp = new byte[6];
78  int count = 0;
79  int digit = 0;
80  tmp[count] = 0;
81  foreach(char ch in hwaddr.ToCharArray()){
82  if((ch == ':') || (ch == '-') || (ch == ' ')){ // delimiter
83  if(++count >= tmp.Length)
84  return this; // Format error
85  tmp[count] = 0;
86  digit = 0;
87  }else{
88  if(digit >= 2){ // delimiter無しで書き連ねた場合
89  if(++count >= tmp.Length)
90  return this; // Too many digits
91  digit = 0;
92  }
93  tmp[count] *= 16;
94  if((ch >= '0') && (ch <= '9'))
95  tmp[count] += (byte)(Convert.ToByte(ch)-Convert.ToByte('0'));
96  else if((ch >= 'a') && (ch <= 'f'))
97  tmp[count] += (byte)(Convert.ToByte(ch)-Convert.ToByte('a')+10);
98  else if((ch >= 'A') && (ch <= 'F'))
99  tmp[count] += (byte)(Convert.ToByte(ch)-Convert.ToByte('A')+10);
100  else
101  return this; // Invalid char for hexadecimal
102  digit++;
103  }
104  }
105  if(digit > 0)
106  count++;
107  vals = new byte[count];
108  Array.Copy(tmp, 0, vals, 0, count);
109  return this;
