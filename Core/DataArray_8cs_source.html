270  /// <summary>
271  /// データ値をコピーする
272  /// </summary>
273  /// <remarks>
274  /// <para>
275  /// 自分自身もコピー元もカラム名を持つとき、同じカラム名のカラムを探して
276  /// 代入が行なわれる。この際、同じカラム名のカラムが存在しない場合は無視
277  /// される。
278  /// 自分自身かコピー元がカラム名を持たないとき、同じカラム番号のデータ値
279  /// をコピーする。この際、コピー元の要素数の方が自分自身の要素数より多い
280  /// 場合は、余った分は無視される。また、コピー元の要素数の方が自分自身の
281  /// 要素数より少ない場合は、足りない分はnullが代入される。
282  /// </para>
283  /// </remarks>
284  public DataArray CopyFrom(DataArray src) {
285  if((columns == null) || (src.columns == null))
286  return CopyFrom(src.Values);
287  lock(mutex) {
288  if(columns == src.columns) {
289  for(int i = 0; i < values.Length; i++)
290  values[i] = src.values[i];
291  return this;
292  }
293  for(int i = 0; i < columns.Length; i++) {
294  for(int j = 0; j < src.columns.Length; j++) {
295  if(columns[i] == src.columns[j]) {
296  values[i] = src.values[j];
297  break;
298  }
299  }
300  }
301  }
302  return this;
303  }

305  /// <summary>
306  /// データ値を代入する
307  /// </summary>
308  /// <remarks>
309  /// <para>
310  /// コピー元の要素数の方が自分自身の要素数より多い場合は、余った分は無視
311  /// される。
312  /// また、コピー元の要素数の方が自分自身の要素数より少ない場合は、足りな
313  /// い分はdefvalが代入される。
314  /// </para>
315  /// </remarks>
316  public DataArray CopyFrom(string[] src, string defval=null) {
317  lock(mutex) {
318  if(src == null) {
319  for(int i = 0; i < values.Length; i++)
320  values[i] = defval;
321  } else {
322  int i = 0;
323  while((i < values.Length) && (i < src.Length)) {
324  values[i] = src[i];
325  i++;
326  }
327  while(i < values.Length) {
328  values[i] = defval;
329  i++;
330  }
331  }
332  }
333  return this;
334  }

359  /// <summary>

361  /// </summary>
362  /// <remarks>
363  /// <para>
364  /// Insert(0, colname, value)で先頭に追加。
365  /// Insert(this.Length, colname, value)はAdd(columnname, value)と等価。
366  /// posがゼロ以下の場合、先頭に追加する。
367  /// posが要素数以上の場合、末尾に追加する。
368  /// カラム名を持たないDataArrayに対しては本関数は利用できない。
369  /// </para>
370  /// </remarks>
371  public DataArray Insert(int pos, string colname, string value) {
372  if(columns == null)
373  throw new ArgumentException("This DataArray doesn't have column names.");
374  lock(mutex) {
375  if(pos < 0)
376  pos = 0;
377  else if(pos > values.Length)
378  pos = values.Length;
379  string[] newcolumns = new string[columns.Length+1];
380  string[] newvalues = new string[values.Length+1];
381  int i = 0;
382  while(i < pos) {
383  newcolumns[i] = columns[i];
384  newvalues[i] = values[i];
385  i++;
386  }
387  newcolumns[i] = colname;
388  newvalues[i] = value;
389  while(i < values.Length) {
390  newcolumns[i+1] = columns[i];
391  newvalues[i+1] = values[i];
392  i++;
393  }
394  columns = newcolumns;
395  values = newvalues;
396  }
397  return this;
398  }

566  /// <summary>
567  /// ストリームを読み取りDataArrayを作る
568  /// </summary>
569  /// <remarks>
570  /// <para>
571  /// カラム名=値 の形式の行を読み取りカラム名付きのDataArrayを生成する。
572  /// カラム名= を持たない行は行番号を文字列にしたダミーのカラム名がつけら
573  /// れる。（カラム名=を持たない最初の行が"0"、次の行が"1"...となる。）
574  /// カラム名および値の前後の空白は除去される。
575  /// 値をダブルクォートもしくはシングルクォートで囲ってもよい。
576  /// 空行は読み飛ばされる。また、"#"で始まる行は読み飛ばされる。
577  /// </para>
578  /// </remarks>
579  public static DataArray From(StreamReader sr) {
580  List<string> columns = new List<string>();
581  List<string> values = new List<string>();
582  int lineno = 0;
583  while(!sr.EndOfStream) {
584  string line = sr.ReadLine().Trim();
585  if((line.Length == 0) || (line[0] == '#'))
586  continue;
587  string key, val;
588  string[] x = line.Split("=".ToCharArray(), 2);
589  if(x.Length == 1) {
590  key = lineno.ToString();
591  val = line;
592  lineno++;
593  } else {
594  key = x[0].Trim();
595  val = x[1].Trim();
596  }
597  if((val.Length >= 2)
598  && (((val[0] == '"') && (val[val.Length-1] == '"'))
599  || ((val[0] == '"') && (val[val.Length-1] == '"')))) {
600  val = val.Substring(1, val.Length-2);
601  }
602  columns.Add(key);
603  values.Add(val);
604  }
605  return new DataArray(columns.ToArray(), values.ToArray());
606  }

642  private DataArray _remove(int pos) {
643  if((pos < 0) || (pos >= values.Length))
644  throw new ArgumentException("Invalid column position for remove.");
645  int i;
646  if(columns != null) {
647  string[] newcolumns = new string[columns.Length-1];
648  i = 0;
649  while(i < pos) {
650  newcolumns[i] = columns[i];
651  i++;
652  }
653  i++;
654  while(i < columns.Length) {
655  newcolumns[i-1] = columns[i];
656  i++;
657  }
658  columns = newcolumns;
659  }
660  string[] newvalues = new string[values.Length-1];
661  i = 0;
662  while(i < pos) {
663  newvalues[i] = values[i];
664  i++;
665  }
