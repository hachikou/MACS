340  private Thread watchdogThread = null;
341  private Stopwatch watchdogTimer = null;
342  private long watchdogTimeout = 1000;
343  private int watchdogCycle = 100;
344  private int watchdogCount = 0;
345  private WatchdogMode watchdogMode = WatchdogMode.LogOnly;
346  private volatile bool isAborting = false;
347 
348  private const int MAX_WATCHDOG_CYCLE = 1000;
349 
350  private void watchdog(long timeout, WatchdogMode mode) {
351  if(watchdogTimer == null) {
352  watchdogTimer = new Stopwatch();
353  }
354  lock(watchdogTimer) {
355  if(timeout > 0) {
356  watchdogTimeout = timeout;
357  // 監視周期はwatchdogTimeout/10。ただし最低1msec、最大1000msec。
358  if(watchdogTimeout < 10)
359  watchdogCycle = 1;
360  else if(watchdogTimeout < (long)MAX_WATCHDOG_CYCLE*10)
361  watchdogCycle = (int)(watchdogTimeout/10L);
362  else
363  watchdogCycle = MAX_WATCHDOG_CYCLE;
364  }
365  watchdogTimer.Restart();
366  if(mode != WatchdogMode.Keep)
367  watchdogMode = mode;
368  watchdogCount = 0;
369  }
370  if(watchdogThread == null) {
371  watchdogThread = new Thread(doWatchdog);
372  watchdogThread.Name = this.Name+"-watchdog";
373  watchdogThread.IsBackground = true;
374  watchdogThread.Start();
375  }
376  }
377 
378  private void stopWatchdog() {
379  if(watchdogThread == null)
380  return;
381  watchdogThread.Abort();
382  watchdogThread = null;
383  if(watchdogTimer != null) {
384  watchdogTimer = null;
385  }
386  }
387 
388  private void doWatchdog() {
389  while((watchdogThread != null) && (watchdogTimer != null)) {
390  if(watchdogTimer.ElapsedMilliseconds >= watchdogTimeout) {
391  watchdogTimer.Restart();
392  watchdogCount++;
393  switch(watchdogMode) {
394  case WatchdogMode.Interrupt:
395  LOG_NOTICE("Watchdog alert on thread {0} ({1}msec), send interrupt", this.Name, watchdogTimeout);
396  watchdogTimeout = long.MaxValue;
397  thread.Interrupt();
398  break;
399  case WatchdogMode.Abort:
400  LOG_NOTICE("Watchdog alert on thread {0} ({1}msec), send abort", this.Name, watchdogTimeout);
401  watchdogTimeout = long.MaxValue;
402  isAborting = true;
403  thread.Abort(new TimeoutException(String.Format("Watchdog alert ({0}msec)", watchdogTimeout)));
404  break;
405  case WatchdogMode.DelayedInterrupt:
406  if(watchdogCount > 1) {
407  LOG_NOTICE("Watchdog alert twice on thread {0} ({1}msec), send interrupt", this.Name, watchdogTimeout);
408  watchdogTimeout = long.MaxValue;
409  thread.Interrupt();
410  } else {
411  LOG_NOTICE("Watchdog alert on thread {0} ({1}msec), will send interrupt on next alert", this.Name, watchdogTimeout);
412  }
413  break;
414  case WatchdogMode.DelayedAbort:
415  if(watchdogCount > 1) {
416  LOG_NOTICE("Watchdog alert twice on thread {0} ({1}msec), send abort", this.Name, watchdogTimeout);
417  watchdogTimeout = long.MaxValue;
418  isAborting = true;
419  thread.Abort(new TimeoutException(String.Format("Watchdog alert ({0}msec)", watchdogTimeout)));
420  } else {
421  LOG_NOTICE("Watchdog alert on thread {0} ({1}msec), will send abort on next alert", this.Name, watchdogTimeout);
422  }
423  break;
424  default:
425  LOG_NOTICE("Watchdog alert on thread {0} ({1}msec)", this.Name, watchdogTimeout);
426  break;
427  }
428  }
429  Thread.Sleep(watchdogCycle);
430  }
431  }
