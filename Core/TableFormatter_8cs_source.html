230  private void Format(Translatable page, StringBuilder sb, string txt, Record rec) {
231  string tt = "";
232  int i,j;
233  switch(dateformat){
234  case DateFormat.SHORT:
235  tt += txt;
236  i = txt.IndexOf('/');
237  j = txt.LastIndexOf(':');
238  if((i >= 0) && (j > i))
239  txt = txt.Substring(i+1,j-i-1).Replace(" ", "&nbsp;");
240  break;
241  case DateFormat.NOYEAR:
242  tt += txt;
243  i = txt.IndexOf('/');
244  if(i >= 0)
245  txt = txt.Substring(i+1).Replace(" ", "&nbsp;");
246  break;
247  case DateFormat.DATEONLY:
248  tt += txt;
249  i = txt.IndexOf(' ');
250  if(i >= 0)
251  txt = txt.Substring(0,i);
252  break;
253  }
254  if(tooltip != null){
255  if(tt != "")
256  tt += ": ";
257  i = rec.GetId(tooltip);
258  if(i >= 0){
259  tt += string.Format(page._("{0}における{1}"), rec.GetString(i), displayname);
260  }
261  }
262  if(tt == ""){
263  sb.Append(txt);
264  }else{
265  sb.Append("<span title=\"");
266  sb.Append(HtmlTool.QE(tt));
267  sb.Append("\">");
268  sb.Append(txt);
269  sb.Append("</span>");
270  }
271  }

553  private bool ParseCellDefs(string txt, bool opflag) {
554  if(celldefs == null)
555  celldefs = new List<List<CellDef>>();
556  foreach(string columntext in txt.Split(",".ToCharArray())){
557  List<CellDef> column = new List<CellDef>();
558  foreach(string celltext in columntext.Split("/".ToCharArray())){
559  string[] def = celltext.Split(":".ToCharArray());
560  CellDef cell = new CellDef();
561  if(def.Length < 2){
562  cell.recordname = "ERROR";
563  cell.recordid = 99;
564  cell.displayname = "ERROR";
565  column.Add(cell);
566  continue;
567  }
568  cell.recordname = def[0];
569  cell.recordid = -1;
570  cell.displayname = def[1];
571  for(int i = 2; i < def.Length; i++){
572  if(def[i].Length == 0)
573  continue;
574  string param = def[i].Substring(1);
575  switch(def[i][0]){
576  case 'W':
577  cell.width = StringUtil.ToInt(param);
578  break;
579  case 'P':
580  cell.decoration = CellDef.Decoration.PAREN;
581  break;
582  case 'G':
583  cell.decoration = CellDef.Decoration.LTGT;
584  break;
585  case 'I':
586  cell.style = CellDef.Style.SUBINFO;
587  break;
588  case 'O':
589  cell.style = CellDef.Style.OPERATION;
590  break;
591  case 'D':
592  if(param.Length == 0)
593  break;
594  switch(param[0]){
595  case 'S':
596  cell.dateformat = CellDef.DateFormat.SHORT;
597  break;
598  case 'N':
599  cell.dateformat = CellDef.DateFormat.NOYEAR;
600  break;
601  case 'D':
602  cell.dateformat = CellDef.DateFormat.DATEONLY;
603  break;
604  }
605  break;
606  case 'K':
607  cell.strikeout = param;
608  break;
609  case 'S':
610  if(param.Length == 0)
611  break;
612  switch(param[0]){
613  case 'A':
614  cell.sortable = CellDef.Sortable.ASCEND;
615  break;
616  case 'D':
617  cell.sortable = CellDef.Sortable.DESCEND;
618  break;
619  }
620  break;
621  case 'T':
622  cell.tooltip = param;
623  break;
624  }
625  }
626  if((cell.style != CellDef.Style.OPERATION) || opflag)
627  column.Add(cell);
628  }
629  if(column.Count > 0)
630  celldefs.Add(column);
631  }
632  return true;
633  }

636  foreach(string txt in strings){
637  if(!ParseCellDefs(txt, opflag))
638  return false;
639  }
