76  /// <summary>
77  /// ファイルをダウンロードする
78  /// </summary>
79  public void Download(string filename, FileStream fs) {
80  Uri u = new Uri("ftp://"+host+"/"+filename);
81  //Console.WriteLine("URI={0}", u.ToString());
82  FtpWebRequest req = (FtpWebRequest)WebRequest.Create(u);
83  if((user != null) && (passwd != null))
84  req.Credentials = new NetworkCredential(user, passwd);
85  req.Method = WebRequestMethods.Ftp.DownloadFile;
86  req.KeepAlive = false;
87  req.UseBinary = true;
88  req.UsePassive = isPassive;
89  //Console.WriteLine("req ok");
90  using(FtpWebResponse res = (FtpWebResponse)(req.GetResponse())) {
91  //Console.WriteLine("res ok");
92  using(Stream sr = res.GetResponseStream()) {
93  //Console.WriteLine("stream ok");
94  byte[] buf = new byte[1024];
95  while(true) {
96  int len = sr.Read(buf, 0, buf.Length);
97  if(len == 0)
98  break;
99  fs.Write(buf, 0, len);
100  }
101  sr.Close();
102  //Console.WriteLine("stream closed");
103  }
104  res.Close();
105  //Console.WriteLine("res closed");
106  }
107  }

144  }

146  using(FtpWebResponse res = (FtpWebResponse)(req.GetResponse())) {
