184  /// <summary>
185  /// "key=value"形式の定義を取り込む
186  /// </summary>
187  public void Set(string str) {
188  int count = 0;
189  StringBuilder key = new StringBuilder();
190  StringBuilder value = new StringBuilder();
191  Status st = Status.BEFORE_KEY;
192  foreach(char ch in str) {
193  switch(st){
194  case Status.BEFORE_KEY:
195  if(ch != ' '){
196  key.Length = 0;
197  key.Append(ch);
198  st = Status.IN_KEY;
199  }
200  break;
201  case Status.IN_KEY:
202  if(ch == ' ')
203  st = Status.BEFORE_EQUAL;
204  else if(ch == '=')
205  st = Status.BEFORE_VALUE;
206  else
207  key.Append(ch);
208  break;
209  case Status.BEFORE_EQUAL:
210  if(ch == '=')
211  st = Status.BEFORE_VALUE;
212  else if(ch != ' '){
213  this[count.ToString()] = ToObject(key.ToString());
214  count++;
215  key.Length = 0;
216  key.Append(ch);
217  st = Status.IN_KEY;
218  }
219  break;
220  case Status.BEFORE_VALUE:
221  if(ch == '"'){
222  value.Length = 0;
223  st = Status.IN_QUOTEVALUE;
224  }else if(ch != ' '){
225  value.Length = 0;
226  value.Append(ch);
227  st = Status.IN_VALUE;
228  }
229  break;
230  case Status.IN_VALUE:
231  if(ch == ' '){
232  this[key.ToString()] = ToObject(value.ToString());
233  st = Status.BEFORE_KEY;
234  }else{
235  value.Append(ch);
236  }
237  break;
238  case Status.IN_QUOTEVALUE:
239  if(ch == '"'){
240  this[key.ToString()] = value.ToString();
241  st = Status.BEFORE_KEY;
242  }else{
243  value.Append(ch);
244  }
245  break;
246  }
247  }
248  switch(st){
249  case Status.IN_KEY:
250  case Status.BEFORE_EQUAL:
251  this[count.ToString()] = ToObject(key.ToString());
252  break;
253  case Status.IN_VALUE:
254  this[key.ToString()] = ToObject(value.ToString());
255  break;
256  case Status.IN_QUOTEVALUE:
257  this[key.ToString()] = value.ToString();
258  break;
259  }
260  }

485  public string ExtractVariables(string str, bool keepUnknownVariable=true, char escChar='\0') {
486  if(String.IsNullOrEmpty(str))
487  return "";
488  StringBuilder ret = new StringBuilder();
489  StringBuilder v = new StringBuilder();
490  bool inEscape = false;
491  bool inVar = false;
492  foreach(char ch in str) {
493  if(!inEscape && (ch == escChar)) {
494  inEscape = true;
495  continue;
496  }
497  if(inVar) {
498  if(!inEscape && (ch == '}')) {
499  ret.Append(ExtractVariableOne(v.ToString(), keepUnknownVariable));
500  v.Clear();
501  inVar = false;
502  } else {
503  v.Append(ch);
504  }
505  } else {
506  if(!inEscape && (ch == '{')) {
507  inVar = true;
508  } else {
509  ret.Append(ch);
510  }
511  }
512  }
513  if(inVar)
514  ret.Append(v.ToString());
515  return ret.ToString();

518  public string ExtractVariableOne(string str, bool keepUnknownVariable=true) {
519  if(String.IsNullOrEmpty(str)) {
520  return keepUnknownVariable?"{}":"";
521  }
522  string[] x = str.Trim().Split(":,".ToCharArray());
523  // x[0]が変数名、x[1...]がフォーマットオプション
524  string vname = x[0].Trim();
525  object val;
526  if(!TryGetValue(vname, out val)) {
527  return keepUnknownVariable?("{"+str+"}"):"";
528  }
529  if((val == null) || (val.ToString() == ""))
530  return "";
531  int len = int.MaxValue;
532  string ret = null;
533  string unit = "";
534  for(int i = 1; i < x.Length; i++) {
535  string opt = x[i].Trim();
536  if(opt == "")
537  continue;
538  switch(opt[0]) {
539  case '0':
540  case '1':
541  case '2':
542  case '3':
543  case '4':
544  case '5':
545  case '6':
546  case '7':
547  case '8':
548  case '9':
549  case '-':
550  // 総桁数指定
551  len = StringUtil.ToInt(opt);
552  break;
553  case 'd':
554  case 'D':
555  case 'x':
556  case 'X':
557  // 整数フォーマット
558  try {
559  ret = StringUtil.ToInt(val.ToString()).ToString(opt);
560  } catch(FormatException) {
561  // just ignore
562  }
563  break;
564  case 'f':
565  case 'F':
566  case 'n':
567  case 'N':
568  case 'c':
569  case 'C':
570  // 実数フォーマット
571  try {
572  ret = StringUtil.ToDouble(val.ToString()).ToString(opt);
573  } catch(FormatException) {
574  // just ignore
575  }
576  break;
577  case '@':
578  // 単位
579  unit = StringUtil.Substring(opt, 1, 9999);
580  break;
581  }
582  }
583  if(ret == null)
584  ret = val.ToString();
585  ret += unit;
586  if(len != int.MaxValue) {
587  if(len >= 0) {
588  ret = StringUtil.FixLength(ret, len);
589  } else {
590  ret = StringUtil.FixLengthRight(ret, len);
591  }
592  }
593  return ret;
594  }

637  private static Regex pat_float = new Regex(@"^[+-]?\d+\.\d*$");
638  private static Regex pat_int = new Regex(@"^[+-]?\d+$");
639  private static Regex pat_hex = new Regex(@"^[+-]?0x([0-9a-fA-F]+)$");
640 
641  private object getObject(string fieldname, object baseobj) {
642  if(String.IsNullOrEmpty(fieldname))
643  return baseobj;
644  Match m;
645  // 定数のチェック
646  if((m = pat_int.Match(fieldname)).Success) {
647  return StringUtil.ToInt(fieldname);
648  } else if((m = pat_float.Match(fieldname)).Success) {
649  return StringUtil.ToDouble(fieldname);
650  } else if((m = pat_hex.Match(fieldname)).Success) {
651  return StringUtil.ToHexInt(m.Groups[1].Value);
652  } else if(fieldname == "true") {
653  return true;
654  } else if(fieldname == "false") {
655  return false;
656  } else if((fieldname.Length > 1) && (((fieldname[0] == '"')&&(fieldname[fieldname.Length-1] == '"')) || ((fieldname[0] == '\'')&&(fieldname[fieldname.Length-1] == '\'')))) {
657  return fieldname.Substring(1, fieldname.Length-2);
658  }
659  // .で分割し、最初のフィールド指定をチェック
660  string varname, indexname;
661  splitField(fieldname, out varname, out fieldname);
662  splitIndex(varname, out varname, out indexname);
663  object index = getObject(indexname, null);
664  object obj = null;
665  if(baseobj == null) {
666  // baseobjがnullの場合、辞書を探す
667  if(!this.ContainsKey(varname))
668  return null;
669  obj = this[varname];
670  } else if(baseobj is DataArray) {
671  // baseobjがDataArrayの場合、カラム名を探す
672  DataArray da = baseobj as DataArray;
673  int idx = da.ColumnNum(varname);
674  if(idx >= 0)
675  return da[idx];
676  } else {
677  // baseobjのメンバを探す
678  foreach(FieldInfo fi in baseobj.GetType().GetFields(BindingFlags.Instance|BindingFlags.Public|BindingFlags.FlattenHierarchy)) {
679  if(fi.Name == varname) {
680  obj = fi.GetValue(baseobj);
681  break;
682  }
683  }
684  if(obj == null) {
685  foreach(PropertyInfo pi in baseobj.GetType().GetProperties(BindingFlags.Instance|BindingFlags.Public|BindingFlags.FlattenHierarchy)) {
686  if(pi.Name == varname) {
687  try {
688  obj = pi.GetValue(baseobj, null);
689  } catch(ArgumentException) {
690  obj = null;
691  }
692  break;
693  }
694  }
695  }
696  }
697  // 添字を展開
698  if(index is int) {
699  if(obj is Array) {
700  try {
701  obj = (obj as Array).GetValue((int)index);
702  } catch(ArgumentException) {

742  /// 文字列を'.'の前と後に分割する
743  /// </summary>
744  private static void splitField(string str, out string varname, out string fieldname) {
745  if(String.IsNullOrEmpty(str)) {
746  varname = "";
747  fieldname = "";
748  }
749  bool inEscape = false;
750  bool inQuote = false;
751  bool inDQuote = false;
752  int blockLevel = 0;
753  int i = 0;
754  while(i < str.Length) {
755  char ch = str[i];
756  if(inEscape) {
757  inEscape = false;
758  } else if(inQuote) {
759  if(ch == '\'')
760  inQuote = false;
761  } else if(inDQuote) {
762  if(ch == '"')
763  inDQuote = false;
764  } else if(ch == '\\') {
765  inEscape = true;
766  } else if(ch == '\'') {
767  inQuote = true;
768  } else if(ch == '"') {
769  inDQuote = true;
770  } else if(ch == '[') {
771  blockLevel++;
772  } else if(blockLevel > 0) {
773  if(ch == ']')
774  blockLevel--;
775  } else if(ch == '.') {
776  varname = str.Substring(0,i);
777  fieldname = str.Substring(i+1);
778  return;
779  }
780  i++;
781  }
782  varname = str;
783  fieldname = "";
784  }
