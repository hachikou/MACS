41  private static void loadElement(XmlElement elem, Element parent, List<Element> elementList) {
42  Element el = new Element(elem, parent, elem.Name.StartsWith("container")?"elem":"");
43  elementList.Add(el);

45  if((node.NodeType != XmlNodeType.Element) || CascadedAttribute.IsSpecialTag(node.Name))
46  continue;
