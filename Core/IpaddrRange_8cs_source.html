125  /// <summary>
126  /// アドレス範囲内のアドレスをアドレスリストに追加する
127  /// </summary>
128  /// <param name="list">このリストに追加する</param>
129  /// <param name="limit">最大件数（負の場合制限無し）</param>
130  public void GetList(List<Ipaddr> list, int limit) {
131  Ipaddr addr = new Ipaddr(StartAddr);
132  while(addr <= EndAddr) {
133  if((limit >= 0) && (list.Count >= limit))
134  return;
135  if(!list.Contains(addr))
136  list.Add(new Ipaddr(addr));
137  addr.Incr();
138  }
139  }
