65  /// <summary>
66  /// 改行まで読み取る。
67  /// </summary>
68  /// <remarks>
69  /// <para>
70  /// 返り値に改行文字は含まない。
71  /// 改行文字はLFまたはCRLF。
72  /// </para>
73  /// </remarks>
74  public string ReadLine() {
75  byte[] buf = new byte[1024];
76  int ptr = 0;
77  int r;
78  while((r = fs.ReadByte()) >= 0) {
79  if((char)r == '\n')
80  break;
81  if((char)r == '\r')
82  continue;
83  if(ptr >= buf.Length) {
84  byte[] tmp = new byte[buf.Length+1024];
85  for(int i = 0; i < ptr; i++)
86  tmp[i] = buf[i];
87  buf = tmp;
88  }
89  buf[ptr] = (byte)r;
90  ptr++;
91  }
92  return SJISDictionary.GetString(buf, 0, ptr);
93  }

95  /// <summary>
96  /// 最大nバイト読み取る。
97  /// </summary>
98  /// <param name="n">最大読み取りバイト数</param>
99  /// <param name="str">読み取った文字列を格納するstring</param>
100  /// <returns>読み取ったバイト数。改行文字はバイト数に含まれない</returns>
101  /// <remarks>
102  /// <para>
103  /// nバイト読み取る前にファイル末尾になるか改行文字が読み取られた場合、
104  /// そこで読み取りを打ち切る。改行文字は結果に含まれない。
105  /// </para>
106  /// </remarks>
107  public int Read(int n, out string str) {
108  byte[] buf = new byte[n];
109  int ptr = 0;
110  int r;
111  while((ptr < n) && ((r = fs.ReadByte()) >= 0)) {
112  if((char)r == '\n')
113  break;
114  if((char)r == '\r')
115  continue;
116  buf[ptr] = (byte)r;
117  ptr++;
118  }
119  str = SJISDictionary.GetString(buf, 0, ptr);
120  return ptr;
121  }

123  /// <summary>
124  /// 固定長フィールドを読み取り、DataArrayとして返す。
125  /// </summary>
126  /// <param name="columns">読み取る項目名一覧（DataArrayのColumnsになる）</param>
127  /// <param name="lengths">columnsの各項目のバイト数</param>
128  /// <param name="readout">指定項目を読み取った後、改行までを読み捨てるかどうか</param>
129  /// <param name="checkcr">改行チェックをするかどうか</param>
130  /// <remarks>
131  /// <para>
132  /// readoutがtrueの場合、指定項目を読み取った後、改行コードまでを読み捨てる。
133  /// readoutがfalseでcheckcrがtrueの場合、指定項目を読み取ったすぐ後に改行コードまたはファイル末尾が無い場合にはIOExceptionを発生する。
134  ///
135  /// 指定項目を読み取っている最中に改行コードまたはファイル末尾になった場合にはIOExceptionを発生する。
136  /// ただし、1項目も読み取らないでファイル末尾になった場合は例外は発生せず、nullを返す。
137  /// </para>
138  /// </remarks>
139  public DataArray GetRecordData(string[] columns, int[] lengths, bool readout=true, bool checkcr=true) {
140  if(columns == null)
141  throw new ArgumentNullException("columns must not be null.");
142  if(lengths == null)
143  throw new ArgumentNullException("lengths must not be null.");
144  if(columns.Length != lengths.Length)
145  throw new ArgumentException("columns.Length and lengths.Length must equal.");
146  int n = 0;
147  for(int i = 0; i < lengths.Length; i++)
148  n += lengths[i];
149  byte[] buf = new byte[n];
150  int ptr = 0;
151  int r = 0;
152  while((ptr < n) && ((r = fs.ReadByte()) >= 0)) {
153  if((char)r == '\n')
154  break;
155  if((char)r == '\r')
156  continue;
157  buf[ptr] = (byte)r;
158  ptr++;
159  }
160  if((ptr == 0) && (r < 0))
161  return null;
162  if(ptr < n)
163  throw new IOException(String.Format("Short line (must be longer than {0} bytes)", n));
164  DataArray rec = new DataArray(columns, null);
165  ptr = 0;
166  for(int i = 0; i < columns.Length; i++) {
167  rec[i] = SJISDictionary.GetString(buf, ptr, lengths[i]);
168  ptr += lengths[i];
169  }
170  if(readout) {
171  ReadLine();
172  } else if(checkcr) {
173  string val;
174  if(Read(1, out val) != 0) {
175  throw new IOException(String.Format("Illegal line length (must be {0} bytes).", n));
176  }
177  }
178  return rec;
179  }
