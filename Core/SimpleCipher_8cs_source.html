50  return "";
51  byte[] buf = new byte[src.Length/2];
52  int ptr = 0;
53  for(int i = 0; i < src.Length/2; i++) {
54  buf[i] = (byte)((hexValue(src[i*2])*16+hexValue(src[i*2+1]))^seed[ptr]);
55  if(++ptr >= seed.Length)
56  ptr = 0;
57  }
58  try {
59  return Encoding.UTF8.GetString(buf);
60  } catch(Exception) {

62  }
